cmake_minimum_required(VERSION 3.16)

project(dottalkpp LANGUAGES CXX)

# -------------------------
# Options / feature toggles
# -------------------------
option(DOTTALK_WITH_INDEX      "Build with xindex and enable index features" ON)
option(DOTTALK_WITH_MEMO       "Build with memo sidecar support"             ON)
option(DOTTALK_BUILD_DEMO      "Build dottalk_demo from src/dli/demo_main.cpp" OFF)
option(DOTTALK_BUILD_PYBIND    "Build Python bindings (stub)"                OFF)
option(DOTTALK_STRICT_WARNINGS "Enable strict warnings"                      ON)
option(DOTTALK_WARN_AS_ERR     "Treat warnings as errors"                    OFF)
option(DOTTALK_WITH_TVISION    "Build BrowseTV TUI using Turbo Vision"       ON)

# ---------------
# Language policy
# ---------------
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Match your existing static CRT choice
if(MSVC)
  set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
endif()

# --------------------
# Helper: warn levels
# --------------------
function(dtt_setup_warnings tgt)
  if(MSVC)
    target_compile_options(${tgt} PRIVATE /permissive- /W4 /EHsc /Zc:__cplusplus)
    if(DOTTALK_STRICT_WARNINGS)
      target_compile_definitions(${tgt} PRIVATE _CRT_SECURE_NO_WARNINGS NOMINMAX WIN32_LEAN_AND_MEAN)
    endif()
    if(DOTTALK_WARN_AS_ERR)
      target_compile_options(${tgt} PRIVATE /WX)
    endif()
  else()
    target_compile_options(${tgt} PRIVATE -Wall -Wextra -Wpedantic)
    if(DOTTALK_WARN_AS_ERR)
      target_compile_options(${tgt} PRIVATE -Werror)
    endif()
  endif()
endfunction()

# ----------------
# Source discovery
# ----------------
set(PROJ_ROOT "${CMAKE_CURRENT_SOURCE_DIR}")
set(INC_DIR   "${PROJ_ROOT}/include")

file(GLOB SRC_XBASE    CONFIGURE_DEPENDS "${PROJ_ROOT}/src/xbase/*.cpp")
file(GLOB SRC_XINDEX   CONFIGURE_DEPENDS "${PROJ_ROOT}/src/xindex/*.cpp")
file(GLOB SRC_MEMO     CONFIGURE_DEPENDS "${PROJ_ROOT}/src/memo/*.cpp")
file(GLOB SRC_CLI      CONFIGURE_DEPENDS "${PROJ_ROOT}/src/cli/*.cpp")
file(GLOB SRC_CLI_EXPR CONFIGURE_DEPENDS "${PROJ_ROOT}/src/cli/expr/*.cpp")
file(GLOB SRC_DLI      CONFIGURE_DEPENDS "${PROJ_ROOT}/src/dli/*.cpp")

# Avoid duplicate ASCEND/DESCEND
list(FILTER SRC_CLI EXCLUDE REGEX "/cmd_order\\.cpp$")

# Keep demo main out of shared DLI list
set(SRC_DLI_CORE ${SRC_DLI})
list(FILTER SRC_DLI_CORE EXCLUDE REGEX "demo_main\\.cpp$")

# Always exclude BrowseTV from the generic CLI list; we add it back only if TVision is found
list(FILTER SRC_CLI EXCLUDE REGEX "/cmd_BROWSETV\\.cpp$")

# --------------------
# Turbo Vision (vcpkg)
# --------------------
# Expect header includes like: #include <tvision/tv.h>
set(TVISION_ENABLED OFF)
if(DOTTALK_WITH_TVISION)
  find_package(tvision CONFIG QUIET)
  if (tvision_FOUND)
    set(TVISION_ENABLED ON)
  else()
    message(STATUS "Turbo Vision (tvision) not found. BrowseTV command will be disabled. "
                   "If you're using vcpkg, pass the toolchain file and enable manifest mode.")
  endif()
endif()

# --------------------
# Libraries
# --------------------
add_library(xbase STATIC ${SRC_XBASE})
target_include_directories(xbase PUBLIC "${INC_DIR}")
target_compile_features(xbase PUBLIC cxx_std_20)
dtt_setup_warnings(xbase)

if(DOTTALK_WITH_INDEX)
  add_library(xindex STATIC ${SRC_XINDEX})
  target_include_directories(xindex PUBLIC "${INC_DIR}")
  target_link_libraries(xindex PUBLIC xbase)
  target_compile_features(xindex PUBLIC cxx_std_20)
  dtt_setup_warnings(xindex)
  target_compile_definitions(xindex PRIVATE DOTTALK_WITH_INDEX=1)
endif()

if(DOTTALK_WITH_MEMO)
  add_library(memo STATIC ${SRC_MEMO})
  target_include_directories(memo PUBLIC "${INC_DIR}")
  target_link_libraries(memo PUBLIC xbase)
  target_compile_features(memo PUBLIC cxx_std_20)
  dtt_setup_warnings(memo)
endif()

# ------------------
# Executable: CLI REPL
# ------------------
add_executable(dottalkpp ${SRC_CLI} ${SRC_CLI_EXPR} ${SRC_DLI_CORE})
target_include_directories(dottalkpp PRIVATE "${INC_DIR}")

# Add BrowseTV only when TVision is present
if (TVISION_ENABLED)
  target_sources(dottalkpp PRIVATE "${PROJ_ROOT}/src/cli/cmd_BROWSETV.cpp")
  target_link_libraries(dottalkpp PRIVATE tvision::tvision)
  target_compile_definitions(dottalkpp PRIVATE DOTTALK_WITH_TVISION=1)
endif()

target_link_libraries(dottalkpp PRIVATE xbase)
if(DOTTALK_WITH_INDEX)
  target_link_libraries(dottalkpp PRIVATE xindex)
endif()
if(DOTTALK_WITH_MEMO)
  target_link_libraries(dottalkpp PRIVATE memo)
endif()

# Only the CLI needs these feature macros; do NOT define them PUBLIC on libs
target_compile_definitions(dottalkpp PRIVATE
  DOTTALK_WITH_INDEX=$<BOOL:${DOTTALK_WITH_INDEX}>
  DOTTALK_WITH_MEMO=$<BOOL:${DOTTALK_WITH_MEMO}>
)

target_compile_features(dottalkpp PRIVATE cxx_std_20)
dtt_setup_warnings(dottalkpp)

# ------------------
# Optional demo exe
# ------------------
if(DOTTALK_BUILD_DEMO)
  add_executable(dottalk_demo "${PROJ_ROOT}/src/dli/demo_main.cpp" ${SRC_DLI_CORE})
  target_include_directories(dottalk_demo PRIVATE "${INC_DIR}")
  target_link_libraries(dottalk_demo PRIVATE xbase)
  if(DOTTALK_WITH_INDEX)
    target_link_libraries(dottalk_demo PRIVATE xindex)
  endif()
  if(DOTTALK_WITH_MEMO)
    target_link_libraries(dottalk_demo PRIVATE memo)
  endif()
  target_compile_definitions(dottalk_demo PRIVATE
    DOTTALK_WITH_INDEX=$<BOOL:${DOTTALK_WITH_INDEX}>
    DOTTALK_WITH_MEMO=$<BOOL:${DOTTALK_WITH_MEMO}>
  )
  target_compile_features(dottalk_demo PRIVATE cxx_std_20)
  dtt_setup_warnings(dottalk_demo)
endif()

# --------------------------
# IDE nicety: source grouping
# --------------------------
set(ALL_SOURCES ${SRC_XBASE} ${SRC_XINDEX} ${SRC_MEMO} ${SRC_CLI} ${SRC_CLI_EXPR} ${SRC_DLI_CORE})
if(EXISTS "${PROJ_ROOT}/src/dli/demo_main.cpp")
  list(APPEND ALL_SOURCES "${PROJ_ROOT}/src/dli/demo_main.cpp")
endif()
# Append TV file to groups if present so it shows in IDE
if (EXISTS "${PROJ_ROOT}/src/cli/cmd_BROWSETV.cpp")
  list(APPEND ALL_SOURCES "${PROJ_ROOT}/src/cli/cmd_BROWSETV.cpp")
endif()
source_group(TREE "${PROJ_ROOT}" FILES ${ALL_SOURCES})

# Summary
# ---------
message(STATUS "=============================================")
message(STATUS " DOTTALK_WITH_INDEX   : ${DOTTALK_WITH_INDEX}")
message(STATUS " DOTTALK_WITH_MEMO    : ${DOTTALK_WITH_MEMO}")
message(STATUS " DOTTALK_WITH_TVISION : ${DOTTALK_WITH_TVISION}")
message(STATUS " TVision (found)      : ${tvision_FOUND}")
message(STATUS " DOTTALK_BUILD_DEMO   : ${DOTTALK_BUILD_DEMO}")
message(STATUS " DOTTALK_BUILD_PYBIND : ${DOTTALK_BUILD_PYBIND}")
message(STATUS " STRICT_WARNINGS      : ${DOTTALK_STRICT_WARNINGS}")
message(STATUS " WARN_AS_ERR          : ${DOTTALK_WARN_AS_ERR}")
message(STATUS " Generator            : ${CMAKE_GENERATOR}")
message(STATUS " CMake version        : ${CMAKE_VERSION}")
message(STATUS "=============================================")
