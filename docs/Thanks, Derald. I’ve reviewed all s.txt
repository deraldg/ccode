// === App.cpp ===
#include <wx/wx.h>
#include "MainWorkspaceFrame.h"

class DottalkApp : public wxApp {
public:
    virtual bool OnInit() {
        MainWorkspaceFrame* frame = new MainWorkspaceFrame();
        frame->Show(true);
        return true;
    }
};

wxIMPLEMENT_APP(DottalkApp);

// === MainWorkspaceFrame.h ===
#pragma once
#include <wx/wx.h>
#include <wx/grid.h>
#include <wx/menu.h>
#include <wx/sizer.h>

class MainWorkspaceFrame : public wxFrame {
public:
    MainWorkspaceFrame();
private:
    void OnOpenDBF(wxCommandEvent& event);
    void OnExit(wxCommandEvent& event);
    void OnVerbSelect(wxCommandEvent& event);
    void OnVerbUse(wxCommandEvent& event);
    void OnVerbArea(wxCommandEvent& event);

    wxGrid* dbfGrid;
    wxStaticText* overlayLabel;
    wxDECLARE_EVENT_TABLE();
};

// === MainWorkspaceFrame.cpp ===
#include "MainWorkspaceFrame.h"

enum {
    ID_OpenDBF = 101,
    ID_VerbSelect,
    ID_VerbUse,
    ID_VerbArea
};

wxBEGIN_EVENT_TABLE(MainWorkspaceFrame, wxFrame)
    EVT_MENU(ID_OpenDBF, MainWorkspaceFrame::OnOpenDBF)
    EVT_MENU(wxID_EXIT, MainWorkspaceFrame::OnExit)
    EVT_MENU(ID_VerbSelect, MainWorkspaceFrame::OnVerbSelect)
    EVT_MENU(ID_VerbUse, MainWorkspaceFrame::OnVerbUse)
    EVT_MENU(ID_VerbArea, MainWorkspaceFrame::OnVerbArea)
wxEND_EVENT_TABLE()

MainWorkspaceFrame::MainWorkspaceFrame()
    : wxFrame(nullptr, wxID_ANY, "dottalk++ Workspace", wxDefaultPosition, wxSize(800, 600)) {

    wxMenuBar* menuBar = new wxMenuBar;
    wxMenu* fileMenu = new wxMenu;
    fileMenu->Append(ID_OpenDBF, "&Open DBF...");
    fileMenu->Append(wxID_EXIT, "E&xit");
    menuBar->Append(fileMenu, "&File");

    wxMenu* verbsMenu = new wxMenu;
    verbsMenu->Append(ID_VerbSelect, "&SELECT");
    verbsMenu->Append(ID_VerbUse, "&USE");
    verbsMenu->Append(ID_VerbArea, "&AREA");
    menuBar->Append(verbsMenu, "&Verbs");

    SetMenuBar(menuBar);

    wxPanel* panel = new wxPanel(this);
    overlayLabel = new wxStaticText(panel, wxID_ANY,
        "Symbolic Overlay: [AREA: 01] [SELECT: active] [USE: dbf_main]", wxPoint(20, 20));

    dbfGrid = new wxGrid(panel, wxID_ANY, wxPoint(20, 60), wxSize(740, 400));
    dbfGrid->CreateGrid(10, 5);
    dbfGrid->SetColLabelValue(0, "ID");
    dbfGrid->SetColLabelValue(1, "Name");
    dbfGrid->SetColLabelValue(2, "Type");
    dbfGrid->SetColLabelValue(3, "Status");
    dbfGrid->SetColLabelValue(4, "Last Modified");

    CreateStatusBar();
    SetStatusText("Ready — AREA 01 active");

    wxBoxSizer* sizer = new wxBoxSizer(wxVERTICAL);
    sizer->Add(overlayLabel, 0, wxALL, 5);
    sizer->Add(dbfGrid, 1, wxEXPAND | wxALL, 5);
    panel->SetSizer(sizer);
}

void MainWorkspaceFrame::OnOpenDBF(wxCommandEvent& event) {
    wxFileDialog openFileDialog(this, _("Open DBF file"), "", "",
        "DBF files (*.dbf)|*.dbf", wxFD_OPEN | wxFD_FILE_MUST_EXIST);

    if (openFileDialog.ShowModal() == wxID_CANCEL)
        return;

    wxString path = openFileDialog.GetPath();
    SetStatusText("Loaded: " + path);
    // TODO: Parse DBF and populate grid
}

void MainWorkspaceFrame::OnExit(wxCommandEvent& event) {
    Close(true);
}

void MainWorkspaceFrame::OnVerbSelect(wxCommandEvent& event) {
    SetStatusText("SELECT verb triggered");
    overlayLabel->SetLabel("Symbolic Overlay: [AREA: 01] [SELECT: triggered] [USE: dbf_main]");
}

void MainWorkspaceFrame::OnVerbUse(wxCommandEvent& event) {
    SetStatusText("USE verb triggered");
    overlayLabel->SetLabel("Symbolic Overlay: [AREA: 01] [SELECT: active] [USE: triggered]");
}

void MainWorkspaceFrame::OnVerbArea(wxCommandEvent& event) {
    SetStatusText("AREA verb triggered");
    overlayLabel->SetLabel("Symbolic Overlay: [AREA: 02] [SELECT: active] [USE: dbf_main]");
}
