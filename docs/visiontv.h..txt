#include <tvision/tv.h>

// Define custom commands
const int cmToggleCmdWin = 1001;  // Custom command for toggling command window size

// Custom Desktop class for blue background with white text
class MyDesktop : public TDeskTop {
public:
    MyDesktop(const TRect& bounds) : TDeskTop(bounds) {}

    TPalette& getPalette() const override {
        static const char cp[] = "\x1F";  // 0x1F: White on blue
        static TPalette p(cp, sizeof(cp) - 1);
        return p;
    }

    void draw() override {
        TDrawBuffer buf;
        ushort color = getColor(1);
        for (int y = 0; y < size.y; y++) {
            buf.fill(0, size.x, ' ', color);  // Fill with space on blue background
            writeBuf(0, y, size.x, 1, buf);
        }
        TDeskTop::draw();
    }
};

// Custom MenuBar class with approximated FoxPro colors
// Normal: white on blue (0x1F), Selected: white on cyan (0x3F), Disabled: yellow on blue (0x1E)
class MyMenuBar : public TMenuBar {
public:
    MyMenuBar(const TRect& bounds, TSubMenu& aMenu) : TMenuBar(bounds, aMenu) {}

    TPalette& getPalette() const override {
        // Approximate structure: 8 for bar, 8 for popup, then more for frames, etc.
        // Using simplified matching to FoxPro W+/B, W+/BG, GR+/B
        static const char cp[] =
            "\x1F\x1F\x1E\x1E\x3F\x3F\x1E\x1E"  // Menu bar: normal text/shortcut, disabled, selected
            "\x1F\x1F\x1E\x1E\x3F\x3F\x1E\x1E"  // Popup menu
            "\x1C\x1F\x3F\x3E\x1F\x3F\x3A\x3B"  // Frames, icons (approx R+/B for frame, etc.)
            "\x39\x2E\x32\x3E\x31\x3F\x3A\x3B"  // Additional (padded to ~48 bytes)
            "\x39\x2E\x32\x3E\x31\x3F\x3A\x3B"
            "\x39\x2E\x32\x3E\x31\x3F\x3A\x3B";
        static TPalette p(cp, sizeof(cp) - 1);
        return p;
    }
};

// Custom StatusLine class with similar colors
class MyStatusLine : public TStatusLine {
public:
    MyStatusLine(const TRect& bounds, TStatusDef& aDefs) : TStatusLine(bounds, aDefs) {}

    TPalette& getPalette() const override {
        static const char cp[] =
            "\x1F\x1F\x1E\x1E\x3F\x3F\x1E\x1E";  // Normal, shortcut, disabled, selected
        static TPalette p(cp, sizeof(cp) - 1);
        return p;
    }
};

// Custom application class
class FoxApp : public TApplication {
private:
    TWindow* cmdWindow;
    bool isFullScreen;

public:
    FoxApp() : TApplication(), isFullScreen(false) {
        // Create and insert the command window (initially 3/5 screen height)
        TRect r = getExtent();
        r.a.y = 1;  // Below menu bar
        int fullHeight = r.b.y - r.a.y;
        r.b.y = r.a.y + (fullHeight * 3 / 5);  // 3/5 height
        cmdWindow = new TEditWindow(r, "Command Window", wnNoNumber);
        cmdWindow->flags |= wfGrow | wfMove | wfZoom;  // Allow resizing
        deskTop->insert(cmdWindow);
    }

    TDeskTop* initDeskTop(TRect r) override {
        return new MyDesktop(r);
    }

    TMenuBar* initMenuBar(TRect r) override {
        r.b.y = r.a.y + 1;
        return new MyMenuBar(r,
            *new TSubMenu("~S~ystem", kbAltS) +  // FoxPro-like menus
                *new TMenuItem("Macros...", cmNo, kbNoKey) +
                *new TMenuItem("Filer", cmNo, kbNoKey) +
                newLine() +
                *new TMenuItem("About", cmAbout, kbNoKey) +

            *new TSubMenu("~F~ile", kbAltF) +
                *new TMenuItem("~O~pen", cmOpen, kbF3, hcNoContext, "F3") +
                *new TMenuItem("~S~ave", cmSave, kbF2, hcNoContext, "F2") +
                *new TMenuItem("Save ~A~s...", cmSaveAs, kbNoKey) +
                newLine() +
                *new TMenuItem("E~x~it", cmQuit, kbAltX, hcNoContext, "Alt-X") +

            *new TSubMenu("~E~dit", kbAltE) +
                *new TMenuItem("~C~ut", cmCut, kbShiftDel, hcNoContext, "Shift+Del") +
                *new TMenuItem("C~o~py", cmCopy, kbCtrlIns, hcNoContext, "Ctrl+Ins") +
                *new TMenuItem("~P~aste", cmPaste, kbShiftIns, hcNoContext, "Shift+Ins") +
                *new TMenuItem("C~l~ear", cmClear, kbNoKey) +

            *new TSubMenu("~D~atabase", kbAltD) +
                *new TMenuItem("Setup...", cmNo, kbNoKey) +
                *new TMenuItem("Append", cmNo, kbNoKey) +
                *new TMenuItem("Pack", cmNo, kbNoKey) +

            *new TSubMenu("~P~rogram", kbAltP) +
                *new TMenuItem("Do...", cmNo, kbNoKey) +
                *new TMenuItem("Compile...", cmNo, kbNoKey) +
                *new TMenuItem("Debug", cmNo, kbNoKey) +

            *new TSubMenu("~W~indow", kbAltW) +
                *new TMenuItem("~T~oggle Command Window", cmToggleCmdWin, kbNoKey) +
                *new TMenuItem("~C~ycle", cmNext, kbF6, hcNoContext, "F6") +
                *new TMenuItem("~S~ize/Move", cmResize, kbCtrlF5, hcNoContext, "Ctrl+F5") +

            *new TSubMenu("~H~elp", kbAltH) +
                *new TMenuItem("~A~bout", cmAbout, kbNoKey) +
                *new TMenuItem("~H~elp Topics", cmHelp, kbF1, hcNoContext, "F1")
        );
    }

    TStatusLine* initStatusLine(TRect r) override {
        r.a.y = r.b.y - 1;
        return new MyStatusLine(r,
            *new TStatusDef(0, 0xFFFF) +
                *new TStatusItem("~Alt-X~ Exit", kbAltX, cmQuit) +
                *new TStatusItem("~F10~ Menu", kbF10, cmMenu) +
                *new TStatusItem(0, kbF1, cmHelp)
        );
    }

    void handleEvent(TEvent& event) override {
        TApplication::handleEvent(event);
        if (event.what == evCommand && event.message.command == cmToggleCmdWin) {
            toggleCmdWin();
            clearEvent(event);
        }
    }

    void toggleCmdWin() {
        TRect r = deskTop->getExtent();
        if (isFullScreen) {
            // Resize to 3/5
            r.b.y = r.a.y + ((r.b.y - r.a.y) * 3 / 5);
            isFullScreen = false;
        } else {
            // Full screen (below menu, above status)
            isFullScreen = true;
        }
        cmdWindow->moveTo(r.a.x, r.a.y);
        cmdWindow->growTo(r.b.x - r.a.x, r.b.y - r.a.y);
        deskTop->redraw();
    }
};

int main() {
    FoxApp app;
    app.run();
    return 0;
}
