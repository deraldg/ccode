## ✅ 1. Replace your old env-directory resolver with this

```cpp
// ============================================================================
// LMDB Environment Directory Resolver (CONTRACT VERSION)
// Contract:
//   <INDEXES>/<DBF_ROOT_UPPER>.cdx   is the LMDB environment directory.
//
// Backward compatibility:
//   If "<ROOT>.cdx.d" exists and "<ROOT>.cdx" does not,
//   use the legacy ".cdx.d" directory.
// ============================================================================

#include <filesystem>
#include <string>

namespace fs = std::filesystem;

static inline bool _ieq(char a, char b)
{
    if (a >= 'A' && a <= 'Z') a = char(a - 'A' + 'a');
    if (b >= 'A' && b <= 'Z') b = char(b - 'A' + 'a');
    return a == b;
}

static inline bool _ends_with_ci(const std::string& s, const std::string& suf)
{
    if (s.size() < suf.size()) return false;
    size_t off = s.size() - suf.size();
    for (size_t i = 0; i < suf.size(); ++i)
        if (!_ieq(s[off + i], suf[i])) return false;
    return true;
}

static inline std::string _dbf_root_upper(std::string token)
{
    fs::path p(token);
    token = p.filename().string();

    if (_ends_with_ci(token, ".dbf"))
        token.resize(token.size() - 4);
    if (_ends_with_ci(token, ".cdx"))
        token.resize(token.size() - 4);

    for (auto& c : token)
        if (c >= 'a' && c <= 'z')
            c = char(c - 'a' + 'A');

    return token;
}

static inline fs::path lmdb_env_dir_from_table(
    const fs::path& indexes_root,
    const std::string& table_token)
{
    const std::string root_up = _dbf_root_upper(table_token);

    fs::path canonical = indexes_root / (root_up + ".cdx");
    fs::path legacy    = indexes_root / (root_up + ".cdx.d");

    if (fs::exists(canonical))
        return canonical;

    if (!fs::exists(canonical) && fs::exists(legacy))
        return legacy; // Backward compatibility

    return canonical;  // Will be created if missing
}
```

## ✅ 2. Replace your old ensure/create directory helper with this

```cpp
// ============================================================================
// LMDB Environment Directory Creation (STRICT CONTRACT)
// ============================================================================

static inline bool ensure_lmdb_env_dir(const fs::path& env_dir, std::string* err)
{
    std::error_code ec;

    if (fs::exists(env_dir, ec))
    {
        if (ec)
        {
            if (err) *err = "Error probing LMDB env path: " + env_dir.string();
            return false;
        }

        if (!fs::is_directory(env_dir, ec))
        {
            if (err) *err =
                "LMDB env path exists but is not a directory: " + env_dir.string();
            return false;
        }

        return true;
    }

    if (!fs::create_directories(env_dir, ec) || ec)
    {
        if (err) *err =
            "Failed to create LMDB env directory: " + env_dir.string();
        return false;
    }

    return true;
}
```
