* LMDB_ORDER_GATHER_01.DTS
* Purpose:
*   Gather evidence about current LMDB/CDX ordered navigation behavior.
*   This script is read-only except for order activation.
*   It avoids DELETE/RECALL/REPLACE/APPEND/PACK.
*
* Focus:
*   - current workspace/schema bring-up
*   - order activation
*   - TOP/BOTTOM/SKIP/LIST/SEEK/FIND agreement
*   - ASCEND/DESCEND behavior
*   - status/area reporting
*
* Notes:
*   - Assumes MCC-style dataset is present
*   - Assumes BUILDLMDB has already been run
*   - Adjust tag names if your current tags differ

ECHO ON
SET TALK ON
SET DELETED ON

ECHO "===== LMDB ORDER GATHER 01 : BEGIN ====="

INIT
SETPATH
WSREPORT ENV

ECHO "===== BRING UP DATASET ====="
SCHEMAS CLOSE
SCHEMAS OPEN DBF
SCHEMAS LOAD mcc.dtschemas
DBAREAS

ECHO "===== AREA / RELATION SNAPSHOT ====="
SELECT STUDENTS
AREA
STATUS
RELATIONS

ECHO "===== NATURAL ORDER BASELINE ====="
SET ORDER TO 0
AREA
TOP
RECORD
SKIP 1
RECORD
SKIP 1
RECORD
BOTTOM
RECORD
LIST 5

ECHO "===== ACTIVATE PRIMARY ORDER : LNAME ====="
SET INDEX TO STUDENTS.CDX
SET ORDER TO TAG LNAME
AREA
STATUS
TOP
RECORD
LIST 5

ECHO "===== STEP THROUGH ACTIVE ORDER ====="
TOP
RECORD
SKIP 1
RECORD
SKIP 1
RECORD
SKIP 1
RECORD

ECHO "===== SEEK / FIND CHECKS ON LNAME ====="
SEEK "SMITH"
RECORD
FIND "SMITH"
RECORD

ECHO "===== DESCENDING CHECK ====="
DESCEND
AREA
TOP
RECORD
LIST 5
SKIP 1
RECORD
SKIP 1
RECORD

ECHO "===== RETURN TO ASCENDING ====="
ASCEND
AREA
TOP
RECORD
LIST 5

ECHO "===== SECOND TAG CHECK : SID ====="
SET ORDER TO TAG SID
AREA
STATUS
TOP
RECORD
LIST 5
SEEK "1"
RECORD

ECHO "===== BACK TO NATURAL ====="
SET ORDER TO 0
AREA
TOP
RECORD
LIST 5

ECHO "===== SECOND TABLE CHECK : TEACHERS / FNAME ====="
SELECT TEACHERS
AREA
STATUS
SET INDEX TO TEACHERS.CDX
SET ORDER TO TAG FNAME
AREA
STATUS
TOP
RECORD
LIST 5
DESCEND
TOP
RECORD
LIST 5
ASCEND

ECHO "===== THIRD TABLE CHECK : DEPT / DEPT_ID ====="
SELECT DEPT
AREA
STATUS
SET INDEX TO DEPT.CDX
SET ORDER TO TAG DEPT_ID
AREA
STATUS
TOP
RECORD
LIST 5
DESCEND
TOP
RECORD
LIST 5
ASCEND

ECHO "===== RAW LMDB/ORDER DIAGNOSTICS ====="
SELECT STUDENTS
AREA
STATUS
LIST_LMDB

ECHO "===== FINAL WORKSPACE SNAPSHOT ====="
DBAREAS
WSREPORT ENV

ECHO "===== LMDB ORDER GATHER 01 : END ====="
