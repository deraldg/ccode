// src/cli/cmd_christmas.cpp
// Animated VT-100-style Christmas tree using dli::screen.

#include <iostream>
#include <sstream>
#include <string>
#include <cstdlib>
#include <chrono>
#include <thread>
#include <vector>

#include "xbase.hpp"
#include "dli/screen.hpp"

static inline int clamp(int v, int lo, int hi) {
    return v < lo ? lo : (v > hi ? hi : v);
}

void cmd_CHRISTMAS(xbase::DbArea& A, std::istringstream& in)
{
    (void)A;
    (void)in;

    using namespace dli;

    const int W = screen_width();
    const int H = screen_height();
    if (W <= 0 || H <= 0) return;

    const bool vt = screen_enable_vt(true);

    // Tree shape (visible chars only).
    const char* tree_lines[] = {
        "    ^",
        "   ^^^",
        "  ^^^^^",
        " ^^^^^^^",
        "^^^^^^^^^",
        "  ^^^^^",
        " ^^^^^^^",
        "^^^^^^^^^"
    };
    const int tree_count = static_cast<int>(sizeof(tree_lines) / sizeof(tree_lines[0]));

    const char ornaments[] = { '*', 'o', '+', '#' };
    const int ornament_count = static_cast<int>(sizeof(ornaments) / sizeof(ornaments[0]));

    const int total_height = tree_count + 3; // star + tree + trunk(2)
    int y0 = (H - total_height) / 2;
    if (y0 < 0) y0 = 0;

    // Simple snow particles.
    struct Snow { int x, y; };
    std::vector<Snow> snow;
    const int snow_count = std::max(10, W / 4);
    snow.reserve(snow_count);
    for (int i = 0; i < snow_count; ++i) {
        snow.push_back(Snow{ std::rand() % W, std::rand() % H });
    }

    const int frames = 60; // ~60 * 150ms ≈ 9 seconds
    const auto frame_delay = std::chrono::milliseconds(150);

    for (int frame = 0; frame < frames; ++frame) {
        screen_clear(false);

        // Star: cycle colors a bit.
        {
            int phase = frame % 4;
            int color = 33; // yellow
            if (phase == 1) color = 37; // white
            else if (phase == 2) color = 36; // cyan
            else if (phase == 3) color = 35; // magenta

            const std::string star = vt_fg("*", color, true, vt);
            const int x = clamp(W / 2, 0, W - 1);
            screen_write_span(x, y0, star);
        }

        // Tree body with twinkling ornaments.
        for (int i = 0; i < tree_count; ++i) {
            const std::string visible(tree_lines[i]);
            const int vis_len = static_cast<int>(visible.size());

            int x = W / 2 - vis_len / 2;
            x = clamp(x, 0, std::max(0, W - vis_len));

            std::string out;
            out.reserve(vis_len * 4);

            for (char c : visible) {
                if (c == '^') {
                    int r = std::rand() % 20;
                    if (r == 0) {
                        std::string s(1, ornaments[std::rand() % ornament_count]);
                        out += vt_fg(s, 31, true, vt); // red
                    } else if (r == 1) {
                        std::string s(1, ornaments[std::rand() % ornament_count]);
                        out += vt_fg(s, 33, true, vt); // yellow
                    } else if (r == 2) {
                        std::string s(1, ornaments[std::rand() % ornament_count]);
                        out += vt_fg(s, 37, false, vt); // white
                    } else {
                        out += vt_fg("^", 32, false, vt); // green
                    }
                } else {
                    out.push_back(c);
                }
            }

            screen_write_span(x, y0 + 1 + i, out);
        }

        // Trunk.
        {
            const std::string trunk_visible = "|||";
            const int vis_len = static_cast<int>(trunk_visible.size());
            int x = W / 2 - vis_len / 2;
            x = clamp(x, 0, std::max(0, W - vis_len));

            const std::string trunk = vt_fg(trunk_visible, 37, false, vt);
            screen_write_span(x, y0 + 1 + tree_count,     trunk);
            screen_write_span(x, y0 + 1 + tree_count + 1, trunk);
        }

        // Snow: draw and advance.
        for (auto& s : snow) {
            screen_write_span(clamp(s.x, 0, W - 1),
                              clamp(s.y, 0, H - 1),
                              vt_fg(".", 37, false, vt));
            s.y += 1;
            if (s.y >= H) {
                s.y = 0;
                s.x = std::rand() % W;
            }
        }

        screen_set_cursor(0, H - 1, false);
        std::this_thread::sleep_for(frame_delay);
    }

    // Leave cursor visible again at bottom.
    screen_set_cursor(0, H - 1, true);
}
