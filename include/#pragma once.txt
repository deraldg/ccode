#pragma once
#include <string>
#include <cctype>
#include <optional>
#include "xbase.hpp"
#include "textio.hpp"

// Case-insensitive, trim-aware field resolver (by name -> 0-based index)
inline int resolve_field_index(xbase::DbArea& db, const std::string& nameIn) {
    const std::string name = textio::trim(nameIn);
    const auto F = db.fields();
    for (int i = 0; i < static_cast<int>(F.size()); ++i) {
        if (textio::ieq(textio::trim(F[i].name), name)) return i;
    }
    return -1;
}

// Standardized getters used by the expr glue macros
inline std::string getFieldAsString(xbase::DbArea& db, const std::string& name) {
    const int idx0 = resolve_field_index(db, name);
    if (idx0 < 0) return std::string{};
    // DbArea::get is 1-based in this codebase; returns display string.
    // (see RecordView::getLogical using a.get(fid0+1))
    std::string s = db.get(idx0 + 1);
    // right-trim spaces from DBF display
    while (!s.empty() && std::isspace(static_cast<unsigned char>(s.back()))) s.pop_back();
    return s;
}

inline double getFieldAsNumber(xbase::DbArea& db, const std::string& name) {
    std::string s = getFieldAsString(db, name);
    // Trim
    auto ltrim = [](std::string& t){
        size_t i=0; while (i<t.size() && std::isspace(static_cast<unsigned char>(t[i]))) ++i; t.erase(0,i);
    };
    ltrim(s);
    // stod throws on bad input; the glue_xbase.hpp catches and returns nullopt.
    size_t pos = 0;
    double v = std::stod(s, &pos);
    if (pos != s.size()) throw std::runtime_error("trailing");
    return v;
}
