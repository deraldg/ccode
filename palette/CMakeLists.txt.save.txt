# File: CMakeLists.txt  (repo root)
cmake_minimum_required(VERSION 3.20)
project(fox_palette_demo LANGUAGES CXX)

# Policies / out-of-source guard
if(POLICY CMP0144)
  cmake_policy(SET CMP0144 NEW)
endif()
if(CMAKE_SOURCE_DIR STREQUAL CMAKE_BINARY_DIR)
  message(FATAL_ERROR "In-source build detected. Configure with -S . -B build-*")
endif()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

option(FOX_USE_ANSI "Use ANSI escape sequences backend" OFF)
set(TVISION_ROOT "C:/Users/deral/code/tvision" CACHE PATH "Local Turbo Vision source tree")
set(DOTTALK_INCLUDE_DIR "" CACHE PATH "Optional: DotTalk++ headers (only if you really want to include them)")

# -------------------- Turbo Vision discovery --------------------
set(_tv_found OFF) ; set(_tv_origin "none")
find_package(tvision CONFIG QUIET)
if(TARGET tvision::tvision AND NOT TARGET tvision)
  add_library(tvision INTERFACE IMPORTED)
  target_link_libraries(tvision INTERFACE tvision::tvision)
endif()
if(TARGET tvision)
  set(_tv_found ON) ; set(_tv_origin "config")
endif()
if(NOT _tv_found AND EXISTS "${TVISION_ROOT}/CMakeLists.txt")
  message(STATUS "Turbo Vision: using local source at ${TVISION_ROOT}")
  add_subdirectory("${TVISION_ROOT}" _tvision_local EXCLUDE_FROM_ALL)
  if(TARGET tvision)
    set(_tv_found ON) ; set(_tv_origin "local")
  endif()
endif()
if(NOT _tv_found AND DEFINED tvision_DIR)
  find_package(tvision CONFIG REQUIRED)
  if(TARGET tvision::tvision AND NOT TARGET tvision)
    add_library(tvision INTERFACE IMPORTED)
    target_link_libraries(tvision INTERFACE tvision::tvision)
  endif()
  if(TARGET tvision)
    set(_tv_found ON) ; set(_tv_origin "explicit-dir")
  endif()
endif()
if(NOT _tv_found)
  message(FATAL_ERROR
    "Turbo Vision NOT found.\n"
    "Try:\n"
    "  A) vcpkg tvision (find_package)\n"
    "  B) -DTVISION_ROOT=C:/path/to/tvision (source)\n"
    "  C) -Dtvision_DIR=C:/path/to/tvision/install/lib/cmake/tvision")
endif()

# -------------------- Core library (no main) --------------------
add_library(fox_palette_core STATIC
  cmd_fox_palette.cpp
  fox_palette.cpp
)
target_include_directories(fox_palette_core PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${DOTTALK_INCLUDE_DIR}  # optional; not required with forward-decls
)
target_link_libraries(fox_palette_core PUBLIC tvision)
if(MSVC)
  target_compile_options(fox_palette_core PRIVATE /W4 /permissive-)
else()
  target_compile_options(fox_palette_core PRIVATE -Wall -Wextra -Wpedantic)
endif()
if(FOX_USE_ANSI)
  target_compile_definitions(fox_palette_core PUBLIC TVISION_USE_ANSI_SEQUENCES)
endif()

# -------------------- Standalone executable --------------------
add_executable(fox_palette
  standalone_main.cpp
)
target_link_libraries(fox_palette PRIVATE fox_palette_core)
# Put exe in build dir
set_target_properties(fox_palette PROPERTIES
  RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}"
  OUTPUT_NAME "fox_palette"
)
if(WIN32)
  if(MSVC)
    target_link_options(fox_palette PRIVATE /SUBSYSTEM:CONSOLE)
  elseif(MINGW)
    target_link_options(fox_palette PRIVATE -Wl,--subsystem,console)
  endif()
endif()

# -------------------- DotTalk++ loadable module (DLL/.so) --------------------
add_library(fox_palette_plugin MODULE
  fox_palette_command.cpp
)
# On Windows we want a .dll (MODULE already does that), ensure name is clean
set_target_properties(fox_palette_plugin PROPERTIES
  OUTPUT_NAME "fox_palette_plugin"
  PREFIX ""  # no "lib" prefix on *nix
  RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}"
  LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}"
  ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}"
)
target_link_libraries(fox_palette_plugin PRIVATE fox_palette_core)
# No subsystem flags for MODULE

install(TARGETS fox_palette RUNTIME DESTINATION bin)
install(TARGETS fox_palette_plugin RUNTIME DESTINATION bin LIBRARY DESTINATION bin)
message(STATUS "Turbo Vision ENABLED via ${_tv_origin}. Targets: exe + plugin + core.")
