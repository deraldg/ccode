# nuke build dir
Remove-Item -Recurse -Force .\build_vs -ErrorAction Ignore

# configure (make sure this is your active vcpkg)
cmake -S . -B build_vs -G "Visual Studio 17 2022" -A x64 `
  -DCMAKE_TOOLCHAIN_FILE="C:/Users/deral/vcpkg/scripts/buildsystems/vcpkg.cmake" `
  -DVCPKG_FEATURE_FLAGS=manifests `
  -DCMAKE_MSVC_RUNTIME_LIBRARY=MultiThreadedDLL `
  -DDOTTALK_WITH_TVISION=ON

# build
cmake --build build_vs --config Release
