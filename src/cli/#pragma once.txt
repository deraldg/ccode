#pragma once
#include <atomic>

namespace dottalk {

// Master switch if you want one later
inline std::atomic<bool> TRACE_ALL{false};

// Fine-grained switches
inline std::atomic<bool> TRACE_LOCATE{false};

// Returns whether a given trace should be active.
// Honors both the specific flag and the global TRACE_ALL.
inline bool trace_locate_active() noexcept {
    return TRACE_ALL.load(std::memory_order_relaxed) ||
           TRACE_LOCATE.load(std::memory_order_relaxed);
}

} // namespace dottalk
