// src/cli/colors.cpp
#include "colors.hpp"

#include <algorithm>
#include <cctype>
#include <iostream>

#if defined(_WIN32)
  #include <windows.h>
#endif

namespace dli { namespace colors {

static std::string up(std::string s) {
    std::transform(s.begin(), s.end(), s.begin(),
                   [](unsigned char c){ return char(std::toupper(c)); });
    return s;
}

Theme parseTheme(const std::string& s) {
    const std::string U = up(s);
    if (U == "GREEN")  return Theme::Green;
    if (U == "AMBER")  return Theme::Amber;
    return Theme::Default;
}

void applyTheme(Theme t) {
#if defined(_WIN32)
    HANDLE h = GetStdHandle(STD_OUTPUT_HANDLE);
    if (h == INVALID_HANDLE_VALUE) return;

    // Default bright gray (console default), Green bright, Amber (yellow) bright
    WORD attr = FOREGROUND_RED | FOREGROUND_GREEN | FOREGROUND_BLUE;
    switch (t) {
      case Theme::Green: attr = FOREGROUND_GREEN | FOREGROUND_INTENSITY; break;
      case Theme::Amber: attr = FOREGROUND_RED | FOREGROUND_GREEN | FOREGROUND_INTENSITY; break;
      case Theme::Default: default: break;
    }
    SetConsoleTextAttribute(h, attr);
#else
    const char* code = "\033[0m";          // default
    switch (t) {
      case Theme::Green: code = "\033[32;1m"; break;   // bright green
      case Theme::Amber: code = "\033[33;1m"; break;   // bright yellow
      case Theme::Default: default: break;
    }
    std::cout << code;
#endif
}

}} // namespace dli::colors
