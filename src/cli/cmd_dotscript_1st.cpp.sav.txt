// src/cli/cmd_dotscript.cpp
// DotScript runner: DOTSCRIPT <file> [QUIET] [STOPONERROR] [ECHO]
// - Reads a text file and feeds each non-blank, non-comment line into the CLI
// - Comments: lines starting with '#', ';', or '//' (after leading whitespace)
// - Uses the existing shell dispatcher to preserve all parsing/behavior

#include <fstream>
#include <sstream>
#include <string>
#include <iostream>
#include <cctype>

#include "xbase.hpp"
#include "textio.hpp"
#include "shell_api.hpp"   // shell_dispatch_line(DbArea&, const std::string&)

// shell_api.cpp exposes this hook; signature verified in your tree:
// bool shell_dispatch_line(DbArea& area, const std::string& line);
// (see: src/cli/shell_api.cpp) :contentReference[oaicite:0]{index=0}

using xbase::DbArea;

namespace {
    inline bool starts_with(const std::string& s, const char* pfx) {
        return s.rfind(pfx, 0) == 0;
    }
    inline bool is_comment_or_blank(const std::string& raw) {
        // Trim left
        size_t i = 0;
        while (i < raw.size() && std::isspace(static_cast<unsigned char>(raw[i]))) ++i;
        if (i >= raw.size()) return true;                 // blank
        if (raw[i] == '#' || raw[i] == ';') return true; // # ; comments
        if (i + 1 < raw.size() && raw[i] == '/' && raw[i+1] == '/') return true; // //
        return false;
    }

    struct Flags {
        bool quiet = false;        // suppress echo of each line
        bool stopOnError = false;  // stop on first failure (unknown command or dispatcher returns false)
        bool echo = false;         // force echo (overrides quiet)
    };

    Flags parse_flags(std::istringstream& iss) {
        Flags f;
        std::string w;
        while (iss >> w) {
            std::string u = textio::up(w);
            if (u == "QUIET")        f.quiet = true;
            else if (u == "STOPONERROR") f.stopOnError = true;
            else if (u == "ECHO")    f.echo = true;
            else {
                // push token back into stream for caller (we advanced one too far)
                // rebuild remaining line starting with this token
                std::string rest;
                std::getline(iss, rest);
                std::string merged = w + rest;
                iss.clear();
                iss.str(merged);
                iss.seekg(0);
                break;
            }
        }
        return f;
    }
} // anon

// DOTSCRIPT <file> [QUIET] [STOPONERROR] [ECHO]
void cmd_DOTSCRIPT(DbArea& A, std::istringstream& iss) {
    if (!A.isOpen()) {
        std::cout << "No table is open. Use USE <file> first (or run a script that opens one).\n";
        // Not fatal; scripts can still run meta commands like DIR, USE, etc.
    }

    // Parse flags (optional) then the file (required). We allow flags before or after file.
    // Strategy: read the entire tail; try to extract a file token possibly quoted.
    std::string tail; std::getline(iss, tail);
    tail = textio::trim(tail);
    if (tail.empty()) {
        std::cout << "Usage: DOTSCRIPT <file> [QUIET] [STOPONERROR] [ECHO]\n";
        return;
    }

    // Support both: DOTSCRIPT file.dts FLAGS...   or   DOTSCRIPT FLAGS... file.dts
    // Try two passes: (1) parse flags then filename; if no filename, (2) assume first token is file then flags.
    std::string filename;
    {
        std::istringstream pass1(tail);
        Flags f = parse_flags(pass1);
        // try to read quoted or bare filename
        std::string tok;
        if (pass1 >> std::ws && pass1.peek() == '"') {
            pass1.get(); // consume leading quote
            std::getline(pass1, tok, '"');
        } else {
            pass1 >> tok;
        }
        if (!tok.empty()) {
            filename = tok;
            // re-parse flags after filename for trailing flags
            std::string after; std::getline(pass1, after);
            std::istringstream pass1b(after);
            Flags f2 = parse_flags(pass1b);
            // combine flags (ECHO beats QUIET if both present)
            f.quiet = (f.quiet || f2.quiet);
            f.stopOnError = (f.stopOnError || f2.stopOnError);
            f.echo = (f.echo || f2.echo);
            if (f.echo) f.quiet = false;

            // store into local static for use below
            // we’ll just capture f by value in the loop.
            // Open and run:
            std::ifstream in(filename);
            if (!in) { std::cout << "DOTSCRIPT: cannot open '" << filename << "'.\n"; return; }

            std::size_t lineNo = 0, ran = 0, skipped = 0, failed = 0;
            std::string line;
            while (std::getline(in, line)) {
                ++lineNo;
                if (is_comment_or_blank(line)) { ++skipped; continue; }
                std::string trimmed = textio::trim(line);
                if (f.echo || (!f.quiet)) {
                    std::cout << "[dts:" << lineNo << "] " << trimmed << "\n";
                }
                // Feed to the existing shell dispatcher
                // Returns false only for QUIT; we treat that as script termination.
                bool ok = shell_dispatch_line(A, trimmed);  // :contentReference[oaicite:1]{index=1}
                if (!ok) { std::cout << "DOTSCRIPT: terminated by QUIT/EXIT at line " << lineNo << ".\n"; break; }

                // shell_dispatch_line will print "Unknown command" itself if not handled.
                // We can’t directly know success of the command; count as ran, not failed.
                ++ran;

                // Optional: minimal heuristic to treat immediate "Unknown command" as failure:
                // (Keeping it simple for now; richer status could be added later via registry.run return plumbing.)
                // If you want STOPONERROR to catch that, we’d need shell_dispatch_line to return an enum.
                // For now STOPONERROR only reacts to QUIT/EXIT (ok=false), which already breaks.
            }

            std::cout << "DOTSCRIPT: file='" << filename << "' ran=" << ran
                      << " skipped=" << skipped << " failed=" << failed << "\n";
            return;
        }
    }

    // Pass 2: assume first token is filename, remaining are flags
    {
        std::istringstream pass2(tail);
        std::string tok;
        if (pass2 >> std::ws && pass2.peek() == '"') {
            pass2.get();
            std::getline(pass2, tok, '"');
        } else {
            pass2 >> tok;
        }
        if (tok.empty()) {
            std::cout << "Usage: DOTSCRIPT <file> [QUIET] [STOPONERROR] [ECHO]\n";
            return;
        }
        filename = tok;
        std::string after; std::getline(pass2, after);
        std::istringstream flagss(after);
        Flags f = parse_flags(flagss);
        if (f.echo) f.quiet = false;

        std::ifstream in(filename);
        if (!in) { std::cout << "DOTSCRIPT: cannot open '" << filename << "'.\n"; return; }

        std::size_t lineNo = 0, ran = 0, skipped = 0, failed = 0;
        std::string line;
        while (std::getline(in, line)) {
            ++lineNo;
            if (is_comment_or_blank(line)) { ++skipped; continue; }
            std::string trimmed = textio::trim(line);
            if (f.echo || (!f.quiet)) {
                std::cout << "[dts:" << lineNo << "] " << trimmed << "\n";
            }
            bool ok = shell_dispatch_line(A, trimmed);      // :contentReference[oaicite:2]{index=2}
            if (!ok) { std::cout << "DOTSCRIPT: terminated by QUIT/EXIT at line " << lineNo << ".\n"; break; }
            ++ran;
        }
        std::cout << "DOTSCRIPT: file='" << filename << "' ran=" << ran
                  << " skipped=" << skipped << " failed=" << failed << "\n";
    }
}

// Self-register like your other handlers
#include "command_registry.hpp"
static bool s_registered = [](){
    cli::registry().add("DOTSCRIPT", &cmd_DOTSCRIPT);
    // friendly aliases
    cli::registry().add("RUN",       &cmd_DOTSCRIPT);
    cli::registry().add("DO",        &cmd_DOTSCRIPT);
    return true;
}();
