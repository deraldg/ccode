// Forward declarations for the core CLI functions
void cmd_USE(xbase::DbArea&, std::istringstream&);
void cmd_LIST(xbase::DbArea&, std::istringstream&);

using namespace DottalkGui;

void DottalkForm::loadButton_Click(System::Object^ sender, System::EventArgs^ e) {
    String^ filename = this->fileTextBox->Text;
    if (String::IsNullOrEmpty(filename)) {
        MessageBox::Show("Please enter a filename.", "Error", MessageBoxButtons::OK, MessageBoxIcon::Error);
        return;
    }

    std::string filenameStd = msclr::interop::marshal_as<std::string>(filename);
    LoadTableToGrid(filenameStd);
}

void DottalkForm::LoadTableToGrid(std::string filename) {
    try {
        // Use a string stream to simulate CLI input
        std::istringstream iss("USE " + filename);
        xbase::DbArea area;

        // Redirect cout to a stringstream to capture output (optional)
        std::stringstream ss;
        std::streambuf* oldCout = std::cout.rdbuf();
        std::cout.rdbuf(ss.rdbuf());

        cmd_USE(area, iss);

        // Restore cout
        std::cout.rdbuf(oldCout);

        // Populate the DataGridView
        mainDataGridView->Rows->Clear();
        mainDataGridView->Columns->Clear();

        // Get the field definitions from the DbArea
        const auto& fields = area.fields();
        for (const auto& field : fields) {
            String^ fieldName = msclr::interop::marshal_as<String^>(field.name);
            mainDataGridView->Columns->Add(fieldName, fieldName);
        }

        // List all records using an istringstream
        std::istringstream listIss("LIST ALL");
        // Re-redirect cout to capture list output
        std::cout.rdbuf(ss.rdbuf());
        cmd_LIST(area, listIss);
        std::cout.rdbuf(oldCout);
        
        // This is a simplistic approach. A better way would be to refactor
        // the core `cmd_list` to return data directly.
        // For this example, we'll manually iterate.

        if (area.recCount() > 0) {
            area.top();
            do {
                if (area.readCurrent()) {
                    System::Windows::Forms::DataGridViewRow^ row = gcnew System::Windows::Forms::DataGridViewRow();
                    row->CreateCells(mainDataGridView);
                    for (int i = 0; i < area.fieldCount(); ++i) {
                        std::string value = area.get(i + 1);
                        row->Cells[i]->Value = msclr::interop::marshal_as<String^>(value);
                    }
                    mainDataGridView->Rows->Add(row);
                }
            } while (area.skip(1));
        }

    } catch (const std::exception& ex) {
        String^ errorMessage = msclr::interop::marshal_as<String^>(ex.what());
        MessageBox::Show("Error: " + errorMessage, "Load Failed", MessageBoxButtons::OK, MessageBoxIcon::Error);
    }
}
