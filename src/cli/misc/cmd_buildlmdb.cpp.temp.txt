// BUILDLMDB / REBUILD for CDX containers.
// Current project direction:
//   - CDX “front end” remains (SET ORDER <table> <tag>).
//   - Backing store becomes LMDB, with one environment per table container.
//   - Multiple tags are stored as named DBs inside that env (MDB_MAXDBS=1024).
//
// Compatibility policy:
//   - If <indexes>/<stem>.cdx exists as a *file*, keep existing RUN1 persistence.
//   - If <indexes>/<stem>.cdx is a *directory*, treat it as LMDB env directory.
//   - If file exists and you still want LMDB, env directory defaults to "<file>.d".

static std::string up_copy(std::string s) {
    for (auto& c : s) c = (char)std::toupper((unsigned char)c);
    return s;
}

static fs::path default_cdx_container_path(xbase::DbArea& area)
{
    std::string stem;
    if (area.isOpen()) {
        stem = area.dbfBasename();
        if (stem.empty()) stem = area.logicalName();
        if (stem.empty()) {
            fs::path n(area.name());
            stem = n.stem().string();
        }
    }
    if (stem.empty()) stem = "table";

    fs::path p = dottalk::paths::resolve_index(stem);
    if (!p.has_extension()) p.replace_extension(".cdx");
    return p;
}

// Decide LMDB environment dir based on the container path.
// If container is already a directory -> use it.
// If container is a file -> use "<file>.d" (safe coexistence).
static fs::path lmdb_env_dir_for_container(const fs::path& cdx_path)
{
    std::error_code ec;
    if (fs::exists(cdx_path, ec) && fs::is_directory(cdx_path, ec)) {
        return cdx_path;
    }
    // file or non-existent: env lives beside/under that name, but cannot collide with a file
    return fs::path(cdx_path.string() + ".d");
}

static bool ensure_dir(const fs::path& p)
{
    std::error_code ec;
    if (fs::exists(p, ec)) return fs::is_directory(p, ec);
    return fs::create_directories(p, ec);
}

// Extract recno from end of composite key: [keybytes ...][uint32_le recno]
static uint32_t rd_u32_le(const unsigned char* b)
{
    return (uint32_t)b[0]
        | ((uint32_t)b[1] << 8)
        | ((uint32_t)b[2] << 16)
        | ((uint32_t)b[3] << 24);
}

// Build a stable composite key.
// NOTE: For now this uses field text bytes padded/truncated to field length (xBase style),
// then appends uint32_le(recno). This preserves (key,recno) uniqueness and stable ordering.
static std::string make_composite_key(const std::string& key_fixed, uint32_t recno)
{
    std::string out = key_fixed;
    unsigned char b[4];
    b[0] = (unsigned char)(recno & 0xFF);
    b[1] = (unsigned char)((recno >> 8) & 0xFF);
    b[2] = (unsigned char)((recno >> 16) & 0xFF);
    b[3] = (unsigned char)((recno >> 24) & 0xFF);
    out.append((const char*)b, 4);
    return out;
}

// Minimal tag build: tag token must be a FIELD NAME in current table.
// This matches what you’ve already been doing for CNX stopgaps.
static bool build_tag_lmdb_from_field(xbase::DbArea& area,
                                     MDB_env* env,
                                     const std::string& tag_name_uc)
{
    if (!area.isOpen()) return false;

    // resolve field index
    int fld = -1;
    const auto& Fs = area.fields();
    for (int i = 0; i < (int)Fs.size(); ++i) {
        if (textio::ieq(Fs[(size_t)i].name, tag_name_uc)) { fld = i + 1; break; }
    }
    if (fld < 1) return false;

    const auto& fdef = Fs[(size_t)(fld - 1)];
    const int keylen = (int)fdef.length;

    // open named DB for the tag
    MDB_txn* txn = nullptr;
    if (mdb_txn_begin(env, nullptr, 0, &txn) != 0) return false;

    MDB_dbi dbi = 0;
    int rc = mdb_dbi_open(txn, tag_name_uc.c_str(), MDB_CREATE, &dbi);
    if (rc != 0) { mdb_txn_abort(txn); return false; }

    // clear existing DB contents (drop+recreate is simplest)
    rc = mdb_drop(txn, dbi, 0);
    if (rc != 0) { mdb_txn_abort(txn); return false; }

    const int32_t total = area.recCount();
    for (int32_t rn = 1; rn <= total; ++rn) {
        if (!area.gotoRec(rn) || !area.readCurrent()) continue;
        if (area.isDeleted()) continue;

        std::string k = area.get(fld);

        // xBase-style: uppercase strings, then fixed-length pad/truncate.
        if (fdef.type == 'C' || fdef.type == 'c') {
            for (char& c : k) c = (char)std::toupper((unsigned char)c);
        }
        if ((int)k.size() > keylen) k.resize((size_t)keylen);
        if ((int)k.size() < keylen) k.append((size_t)(keylen - (int)k.size()), ' ');

        std::string comp = make_composite_key(k, (uint32_t)rn);

        MDB_val mkey{ (size_t)comp.size(), (void*)comp.data() };
        MDB_val mval{ 0, nullptr }; // value not needed; ordering is in key

        rc = mdb_put(txn, dbi, &mkey, &mval, 0);
        if (rc != 0) { mdb_txn_abort(txn); return false; }
    }

    rc = mdb_txn_commit(txn);
    return (rc == 0);
}

static bool build_lmdb_env_for_cdx(xbase::DbArea& area,
                                  const fs::path& cdx_container,
                                  std::vector<std::string>* out_tags_built)
{
    if (out_tags_built) out_tags_built->clear();

    const fs::path envdir = lmdb_env_dir_for_container(cdx_container);
    if (!ensure_dir(envdir)) {
        std::cout << "BUILDLMDB: unable to create env dir: " << envdir.string() << "\n";
        return false;
    }

    MDB_env* env = nullptr;
    int rc = mdb_env_create(&env);
    if (rc != 0 || !env) return false;

    // Your settled decision:
    // max dbs = 1024
    (void)mdb_env_set_maxdbs(env, 1024);

    // Reasonable map size default; adjust later.
    // (You can promote this to SET/CFG once you measure.)
    (void)mdb_env_set_mapsize(env, 256ULL * 1024ULL * 1024ULL);

    // Open in the env directory. (LMDB uses data.mdb/lock.mdb inside that dir.)
    rc = mdb_env_open(env, envdir.string().c_str(), 0, 0664);
    if (rc != 0) {
        mdb_env_close(env);
        std::cout << "BUILDLMDB: mdb_env_open failed: " << rc << "\n";
        return false;
    }

    // Tag discovery:
    // If you still have a file-based CDX header/tagdir, read tags from it.
    // If the container is directory-only, this will be metadata-driven later.
    std::vector<cdxfile::TagInfo> tags;

    std::error_code ec;
    if (fs::exists(cdx_container, ec) && fs::is_regular_file(cdx_container, ec)) {
        cdxfile::CDXHandle* h = nullptr;
        if (cdxfile::open(cdx_container.string(), h)) {
            (void)cdxfile::read_tagdir(h, tags);
            cdxfile::close(h);
        }
    }

    if (tags.empty()) {
        // Fallback: if no tagdir exists yet, just build the currently active tag (if any).
        const std::string active = up_copy(orderstate::activeTag(area));
        if (!active.empty()) tags.push_back(cdxfile::TagInfo{1, active, 0, 0});
    }

    int ok = 0;
    for (const auto& t : tags) {
        const std::string tag_uc = up_copy(t.name);
        if (tag_uc.empty()) continue;

        if (build_tag_lmdb_from_field(area, env, tag_uc)) {
            ++ok;
            if (out_tags_built) out_tags_built->push_back(tag_uc);
        }
    }

    mdb_env_close(env);
    return ok > 0;
}

} // namespace

// Command entrypoint
void cmd_BUILDLMDB(xbase::DbArea& area, std::istringstream& args)
{
    (void)args;

    if (!area.isOpen()) {
        std::cout << "BUILDLMDB: No table open.\n";
        return;
    }

    // Determine the CDX container name for this table.
    // If the user currently has an active CDX order, prefer that.
    fs::path cdx_container;
    if (orderstate::hasOrder(area) && orderstate::isCdx(area)) {
        cdx_container = fs::path(orderstate::orderName(area));
    } else {
        cdx_container = default_cdx_container_path(area);
    }

    std::cout << "REBUILD: " << cdx_container.string() << "\n";

    // If it’s a directory container OR you want LMDB even when a file exists,
    // build LMDB env now.
    // (You can add a CLI switch later: BUILDLMDB LMDB / BUILDLMDB FILE, etc.)
    std::vector<std::string> built;
    if (build_lmdb_env_for_cdx(area, cdx_container, &built)) {
        for (size_t i = 0; i < built.size(); ++i) {
            std::cout << "  [" << (i + 1) << "] " << built[i] << " : OK\n";
        }
        std::cout << "REBUILD: done  OK=" << built.size() << "  SKIP=0  FAIL=0\n";
        std::cout << "REBUILD: LMDB updated (env per table; named DB per tag; maxdbs=1024).\n";
        return;
    }

    // Otherwise fall back to existing file-based RUN1 persistence (your current working path).
    // ---- Existing behavior preserved below (calls your current CDX backend) ----
    {
        // NOTE: This requires your existing backend headers/types. Keep your original code here.
        // If you want me to merge your exact old implementation back in, paste your current
        // cmd_buildlmdb.cpp from the repo and I’ll splice it cleanly.
        std::cout << "REBUILD: LMDB build failed or no tags; file-based rebuild path not inlined here.\n";
    }
}
