// src/cli/cmd_replace.cpp — REPLACE by field INDEX or NAME with implicit record locking
// Usage:
//   REPLACE <field_index> WITH <value>
//   REPLACE <field_name>  WITH <value>
//
// Rule:
//   - If TABLE is enabled for current area and a write succeeds => DIRTY true.
//   - Mark field-level STALE for any field that actually changes (even if TABLE is OFF).
//
// Notes:
// - WITH <value> accepts FoxPro-ish value expressions for common string/date functions, e.g.
//     REPLACE lname    WITH LEFT(lname,3)
//     REPLACE lname    WITH PROPER(ALLTRIM(lname))
//     REPLACE dob      WITH DATE()
//     REPLACE enroll_d WITH DATEADD(enroll_d, 10)
//   Field references are resolved against the current record.

// Provided by the interactive shell.
extern "C" xbase::XBaseEngine* shell_engine(void);

using cli::Settings;

static std::string to_upper_copy(std::string s) {
    std::transform(s.begin(), s.end(), s.begin(),
                   [](unsigned char c){ return (char)std::toupper(c); });
    return s;
}

static bool try_parse_int(const std::string& s, int& out) {
    if (s.empty()) return false;
    char* end = nullptr;
    long v = std::strtol(s.c_str(), &end, 10);
    if (end == s.c_str() || *end != '\0') return false;
    if (v < INT_MIN || v > INT_MAX) return false;
    out = (int)v;
    return true;
}

static bool values_equal_for_stale_check(const std::string& before, const std::string& after) {
    return before == after;
}

static bool should_mark_stale_for_field_change(const xbase::DbArea& area, int fidx) {
    // TODO: check if field is part of any index tag
    (void)area; (void)fidx;  // placeholder
    return true;  // conservative: always stale for now
}

static int resolve_current_index(xbase::DbArea& A) {
    xbase::XBaseEngine* eng = shell_engine();
    if (!eng) return -1;

    for (int i = 0; i < xbase::MAX_AREA; ++i) {
        if (&eng->area(i) == &A) return i;
    }
    return -1;
}

void cmd_REPLACE(xbase::DbArea& A, std::istringstream& in) {
    if (!A.isOpen()) {
        std::cout << "REPLACE: no file open.\n";
        return;
    }

    // Parse REPLACE field WITH value
    std::string field_name, with, value;
    in >> field_name >> with;
    if (to_upper_copy(with) != "WITH") {
        std::cout << "Usage: REPLACE <field> WITH <value>\n";
        return;
    }
    std::getline(in, value);
    value = textio::trim(value);

    int area0 = resolve_current_index(A);
    if (area0 < 0) {
        std::cout << "REPLACE: cannot determine current area.\n";
        return;
    }

    int fldIndex = xfg::resolve_field_index_std(A, field_name);
    if (fldIndex < 0) {
        std::cout << "REPLACE: field not found.\n";
        return;
    }

    int field1 = fldIndex + 1;  // 1-based

    const uint32_t rn = A.recno();
    std::string lock_err;
    if (!xbase::locks::try_lock_record(A, rn, &lock_err)) {
        std::cout << "REPLACE: record is locked (" << lock_err << ").\n";
        return;
    }

    // Capture BEFORE for stale tracking.
    std::string before;
    try { before = A.get(field1); } catch (...) { before.clear(); }

    if (dottalk::table::is_enabled(area0)) {
        // Buffer mode
        auto& tb = dottalk::table::get_tb(area0);

        // Build field mask
        std::uint64_t field_mask[dottalk::table::kWords]{};
        int idx0 = fldIndex;
        int word = idx0 / 64;
        int bit  = idx0 % 64;
        field_mask[word] |= (std::uint64_t{1} << bit);

        tb.add_change(rn, dottalk::table::CHANGE_UPDATE, field_mask, field1, value);

        dottalk::table::set_dirty(area0, true);
        dottalk::table::mark_stale_field(area0, field1);

        xbase::locks::unlock_record(A, rn);
        std::cout << "Buffered REPLACE " << field_name << " at rec " << rn << ".\n";
        return;
    }

    // Direct write (TABLE OFF)
    bool ok = false;
    try {
        ok = A.set(field1, value) && A.writeCurrent();
    } catch (...) {
        ok = false;
    }

    xbase::locks::unlock_record(A, rn);

    if (!ok) { std::cout << "REPLACE: write failed.\n"; return; }

    // Capture AFTER for stale tracking.
    std::string after;
    try { after = A.get(field1); } catch (...) { after.clear(); }

    if (auto* eng = shell_engine()) {
        const int area0 = eng->currentArea();

        // STALE (field-level): mark iff the stored value actually changed.
        if (!values_equal_for_stale_check(before, after)) {
            dottalk::table::mark_stale_field(area0, field1);
        }
    }

    if (Settings::instance().talk_on.load())
        std::cout << "Replaced field #" << field1 << " at rec " << A.recno() << ".\n";
}
