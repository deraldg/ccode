static xbase::DbArea* current_area() {
    try { return const_cast<xbase::DbArea*>(workareas::at(workareas::current_slot())); }
    catch (...) { return nullptr; }
}

static std::string trim(std::string s) {
    auto issp = [](unsigned char c){ return c==' '||c=='\t'||c=='\r'||c=='\n'; };
    while(!s.empty() && issp((unsigned char)s.front())) s.erase(s.begin());
    while(!s.empty() && issp((unsigned char)s.back())) s.pop_back();
    return s;
}
static std::string up(std::string s) { for (auto& c : s) c = (char)std::toupper((unsigned char)c); return s; }

static std::string current_order_hint() {
    auto* A = current_area();
    if (!A) return "physical";
    try {
        if (!orderstate::hasOrder(*A)) return "physical";
        std::ostringstream oss;
        if (orderstate::isCnx(*A)) oss << "CNX:" << orderstate::activeTag(*A);
        else oss << "INX:" << orderstate::orderName(*A);
        oss << (orderstate::isAscending(*A) ? " ASC" : " DESC");
        return oss.str();
    } catch (...) { return "physical"; }
}

static void print_tuple_row(const dottalk::TupleRow& r) {
    for (size_t i = 0; i < r.columns.size() && i < r.values.size(); ++i) {
        if (i) std::cout << " | ";
        std::cout << r.columns[i].name << "=" << (r.values[i].empty() ? "\"\"" : r.values[i]);
    }
    std::cout << "\n";
    if (!r.fragments.empty()) {
        std::cout << "  ";
        for (size_t k = 0; k < r.fragments.size(); ++k) {
            const auto& f = r.fragments[k];
            std::string logical = "(area?)";
            try {
                const xbase::DbArea* A = workareas::at(f.area_slot);
                if (A) {
                    const std::string ln = A->logicalName();
                    logical = ln.empty() ? A->name() : ln;
                }
            } catch (...) {}
            if (k) std::cout << "  ";
            std::cout << "[AREA=" << logical << " RECNO=" << f.recno << "]";
        }
        std::cout << "\n";
    }
}

static void show_area_banner() {
    auto* A = current_area();
    if (!A) { std::cout << "AREA: (none)\n"; return; }
    try {
        std::cout << "AREA:\n";
        std::cout << "  logical:  " << A->logicalName()  << "\n";
        std::cout << "  basename: " << A->dbfBasename()  << "\n";
        std::cout << "  file:     " << A->filename()     << "\n";
        std::cout << "  order:    " << current_order_hint() << "\n";
    } catch (...) { std::cout << "AREA: (unavailable)\n"; }
}

static void show_relations_banner() {
    try {
        const auto v = relations_status::relation_stats_for_current_parent();
        std::cout << "RELATIONS:\n";
        if (v.empty()) { std::cout << "  (none)\n"; return; }
        for (const auto& s : v) std::cout << "  " << s.child_area << ": " << s.match_count << "\n";
    } catch (...) { std::cout << "RELATIONS: (unavailable)\n"; }
}

enum class PagerAction { NextPage, Quit, Repaint };
struct PagerState { bool show_schema=false, show_json=false, status_verbose=false; };
struct PromptResult { PagerAction action; bool nav_event; };

static PromptResult pager_prompt_and_dispatch(dottalk::DbTupleStream& stream, PagerState& ps) {
    std::cout
        << "-- More -- (Enter=next, Q=quit, TOP/BOTTOM/SKIP n/GOTO n, FOR <expr>, CLEAR FOR, ORDER PHYSICAL/INX/CNX, "
        << "SPEC <spec>, SHOW|HIDE SCHEMA|JSON, SHOW AREA, SHOW RELATIONS, SHOW CHILDREN [LIMIT n], SHOW CHILDREN FOR <area> PREVIEW n, OPEN CHILD <area>, SHOW FILTER, SET COLLATION CI|CS, STATUS VERBOSE|COMPACT, HELP) ";
    std::string line; if (!std::getline(std::cin, line)) return {PagerAction::Quit, false};
    const std::string raw = line;
    auto trim_copy = [](std::string s){ auto sp=[](unsigned char c){return c==' '||c=='\t'||c=='\r'||c=='\n';};
        while(!s.empty()&&sp((unsigned char)s.front())) s.erase(s.begin());
        while(!s.empty()&&sp((unsigned char)s.back())) s.pop_back(); return s; };
    line = trim_copy(line);
    if (line.empty()) return {PagerAction::NextPage, false};
    std::string u = up(line);
    if (u == "Q" || u == "QUIT") return {PagerAction::Quit, false};

    if (u == "HELP") {
        std::cout
          << "Super Browser Help\n"
          << "Commands:\n"
          << "  Enter                Next page\n"
          << "  Q|QUIT               Exit browser\n"
          << "  TOP|BOTTOM           Jump to first/last\n"
          << "  SKIP n               Move by n (signed)\n"
          << "  GOTO n               Go to logical position or RECNO (physical)\n"
          << "  ORDER PHYSICAL       Use physical order\n"
          << "  ORDER INX|CNX        Use current INX/CNX order\n"
          << "  FOR <expr>           Set tuple-level filter (AST engine)\n"
          << "  CLEAR FOR            Clear filter\n"
          << "  SPEC <tuple-spec>    Change tuple spec (e.g., TEACHERS.*)\n"
          << "  SHOW SCHEMA          Show logical schema panel\n"
          << "  SHOW JSON            Show JSON/sidecar panel\n"
          << "  SHOW RELATIONS       Show relation summary panel\n"
          << "  SHOW CHILDREN [LIMIT n]                Show child areas with match counts\n"
          << "  SHOW CHILDREN FOR <area> PREVIEW n     Preview n child tuples via tuple builder\n"
          << "  OPEN CHILD <area>                       Drill into child as new parent (spec <area>.*)\n"
          << "  SHOW FILTER          Show current FOR expression\n"
          << "  STATUS VERBOSE|COMPACT  Toggle header detail\n"
          << "  SET COLLATION CI|CS  Case-insensitive/sensitive string compare\n";
        return {PagerAction::Repaint, false};
    }

    if (u == "TOP") { stream.top(); return {PagerAction::Repaint, true}; }
    if (u == "BOTTOM") { stream.bottom(); return {PagerAction::Repaint, true}; }

    if (u == "CLEAR FOR" || u == "CLEARFOR") { stream.set_filter_for(""); stream.top(); return {PagerAction::Repaint, true}; }

    if (u == "SHOW AREA") { show_area_banner(); return {PagerAction::Repaint, false}; }
    if (u == "SHOW RELATIONS") { show_relations_banner(); return {PagerAction::Repaint, false}; }

    if (u.rfind("SHOW CHILDREN FOR ", 0) == 0) {
        // SHOW CHILDREN FOR <area> PREVIEW n
        std::string rest = trim_copy(raw.substr(std::string("SHOW CHILDREN FOR ").size()));
        auto pos_prev = rest.find("PREVIEW");
        if (pos_prev == std::string::npos) { std::cout << "Usage: SHOW CHILDREN FOR <area> PREVIEW n\n"; return {PagerAction::Repaint, false}; }
        std::string area = trim_copy(rest.substr(0, pos_prev));
        int limit = 5; try { std::string ns = trim_copy(rest.substr(pos_prev + 7)); limit = std::stoi(ns); } catch (...) {}
        if (limit < 1) limit = 1; if (limit > 100) limit = 100;
        try {
            relations_api::refresh_for_current_parent();
            auto rows = relations_api::preview_child(area, limit);
            if (rows.empty()) std::cout << "CHILDREN PREVIEW (" << area << "): (none)\n";
            else { std::cout << "CHILDREN PREVIEW (" << area << "):\n"; for (const auto& r : rows) std::cout << "  " << r.line << "\n"; }
        } catch (...) { std::cout << "CHILDREN PREVIEW: (unavailable)\n"; }
        return {PagerAction::Repaint, false};
    }

    if (u.rfind("SHOW CHILDREN", 0) == 0) {
        int limit = 10; auto pos = raw.find("LIMIT");
        if (pos != std::string::npos) { try { limit = std::stoi(raw.substr(pos + 5)); } catch (...) {} if (limit < 1) limit = 1; if (limit > 100) limit = 100; }
        try {
            relations_api::refresh_for_current_parent();
            const auto stats = relations_status::relation_stats_for_current_parent();
            std::cout << "CHILDREN (by relation):\n";
            if (stats.empty()) std::cout << "  (none)\n";
            else { int shown = 0; for (const auto& s : stats) { if (shown++ >= limit) break; std::cout << "  " << s.child_area << " — matches: " << s.match_count << "\n"; } }
        } catch (...) { std::cout << "CHILDREN: (unavailable)\n"; }
        return {PagerAction::Repaint, false};
    }

    if (u.rfind("STATUS ", 0) == 0) {
        const std::string mode = up(trim_copy(u.substr(7)));
        if (mode == "VERBOSE") ps.status_verbose = true;
        else if (mode == "COMPACT") ps.status_verbose = false;
        else std::cout << "SUPERBROWSER: STATUS VERBOSE|COMPACT\n";
        return {PagerAction::Repaint, false};
    }

    if (u == "SHOW FILTER") { std::cout << "FILTER: " << stream.current_filter() << "\n"; return {PagerAction::Repaint, false}; }

    if (u.rfind("SHOW ", 0) == 0) {
        const std::string what = up(trim_copy(u.substr(5)));
        if (what == "SCHEMA") ps.show_schema = true;
        else if (what == "JSON") ps.show_json = true;
        else std::cout << "SUPERBROWSER: unknown SHOW target.\n";
        return {PagerAction::Repaint, false};
    }
    if (u.rfind("HIDE ", 0) == 0) {
        const std::string what = up(trim_copy(u.substr(5)));
        if (what == "SCHEMA") ps.show_schema = false;
        else if (what == "JSON") ps.show_json = false;
        else std::cout << "SUPERBROWSER: unknown HIDE target.\n";
        return {PagerAction::Repaint, false};
    }

    if (u.rfind("SET COLLATION ", 0) == 0) {
        const std::string mode = up(trim_copy(u.substr(14)));
        if (mode == "CI") { dottalk::exprglue::set_collation_ci(true); std::cout << "COLLATION: case-insensitive\n"; }
        else if (mode == "CS") { dottalk::exprglue::set_collation_ci(false); std::cout << "COLLATION: case-sensitive\n"; }
        else std::cout << "SET COLLATION CI|CS\n";
        return {PagerAction::Repaint, false};
    }

    if (u.rfind("ORDER ", 0) == 0) {
        const std::string arg = trim_copy(u.substr(6));
        if (arg == "PHYSICAL") { stream.set_order_physical(); stream.top(); return {PagerAction::Repaint, true}; }
        if (arg == "INX" || arg == "CNX") {
            xbase::DbArea* A = current_area();
            if (!A || !orderstate::hasOrder(*A)) { std::cout << "SUPERBROWSER: no active order. Use SET ORDER/SET INDEX first.\n"; return {PagerAction::Repaint, false}; }
            const bool is_cnx = orderstate::isCnx(*A);
            if (arg == "INX" && is_cnx) { std::cout << "SUPERBROWSER: current order is CNX; expected INX.\n"; return {PagerAction::Repaint, false}; }
            if (arg == "CNX" && !is_cnx) { std::cout << "SUPERBROWSER: current order is INX; expected CNX.\n"; return {PagerAction::Repaint, false}; }
            if (arg == "INX") stream.set_order_inx(); else stream.set_order_cnx();
            stream.top(); return {PagerAction::Repaint, true};
        }
        std::cout << "SUPERBROWSER: unknown ORDER option.\n";
        return {PagerAction::Repaint, false};
    }

    if (u.rfind("OPEN CHILD ", 0) == 0) {
        std::string area = trim_copy(raw.substr(std::string("OPEN CHILD ").size()));
        if (area.empty()) { std::cout << "Usage: OPEN CHILD <area>\n"; return {PagerAction::Repaint, false}; }
        stream.set_spec(area + ".*"); // read-only drill
        stream.top();
        return {PagerAction::Repaint, true};
    }

    if (u.rfind("SPEC ", 0) == 0) {
        std::string ns = trim_copy(raw.substr(5));
        if (ns.empty()) std::cout << "SPEC <tuple-spec>\n";
        else stream.set_spec(ns);
        return {PagerAction::Repaint, true};
    }

    if (u.rfind("SKIP ", 0) == 0) {
        try { long n = std::stol(trim_copy(raw.substr(5))); stream.skip(n); }
        catch (...) { std::cout << "SUPERBROWSER: SKIP expects an integer.\n"; }
        return {PagerAction::Repaint, true};
    }

    if (u.rfind("GOTO ", 0) == 0) {
        try {
            long n = std::stol(trim_copy(raw.substr(5)));
            bool ok = stream.is_ordered() ? stream.goto_pos(n) : stream.goto_recno(n);
            if (!ok) std::cout << "SUPERBROWSER: GOTO out of range.\n";
        } catch (...) { std::cout << "SUPERBROWSER: GOTO expects an integer.\n"; }
        return {PagerAction::Repaint, true};
    }

    if (u.rfind("FOR ", 0) == 0) {
        std::string expr = trim_copy(raw.substr(4));
        stream.set_filter_for(expr);
        stream.top();
        if (auto h = dottalk::exprglue::pop_ambiguity_hint()) {
            std::cout << "WARNING: '" << *h << "' is ambiguous; use AREA.FIELD\n";
        }
        return {PagerAction::Repaint, true};
    }

    std::cout << "(hint: TOP/BOTTOM/SKIP n/GOTO n/FOR <expr>/CLEAR FOR/ORDER .../SPEC <spec>/SHOW|HIDE SCHEMA|JSON/SHOW AREA/SHOW RELATIONS/SHOW CHILDREN[/FOR ... PREVIEW n]/OPEN CHILD/SHOW FILTER/SET COLLATION/STATUS VERBOSE|COMPACT/HELP)\n";
    return {PagerAction::Repaint, false};
}

static void run_super_browser(std::istringstream& iss) {
    std::string spec = "*"; int page_size = 20; std::string for_expr; PagerState ps;

    std::string rest; std::getline(iss, rest);
    auto upper = rest; for (auto& c : upper) c = (char)std::toupper((unsigned char)c);
    auto pos_for = upper.find(" FOR ");
    if (pos_for != std::string::npos) { for_expr = rest.substr(pos_for + 5); rest.erase(pos_for); upper.erase(pos_for); }
    std::istringstream ss(rest);
    std::string tok; bool spec_set=false;
    while (ss >> tok) {
        std::string U = tok; for (auto& c : U) c=(char)std::toupper((unsigned char)c);
        if (U == "PAGESIZE") { ss >> page_size; continue; }
        if (U == "SHOW") { std::string w; ss >> w; std::string W=w; for(auto& c:W)c=(char)std::toupper((unsigned char)c); if(W=="SCHEMA") ps.show_schema=true; else if(W=="JSON") ps.show_json=true; continue; }
        if (U == "STATUS") { std::string w; ss >> w; std::string W=w; for(auto& c:W)c=(char)std::toupper((unsigned char)c); ps.status_verbose=(W=="VERBOSE"); continue; }
        if (!spec_set) { spec_set=true; std::string tail; std::getline(ss, tail); spec = trim(tok + tail); break; }
    }
    if (page_size <= 0) page_size = 20; if (page_size > 200) page_size = 200;

    dottalk::DbTupleStream stream(spec, "");
    if (!for_expr.empty()) stream.set_filter_for(for_expr);
    stream.top();

    if (!current_area()) {
        std::cout << "Super Browser: no current AREA.\nHint: USE <table> or SELECT <n> (alias) before running sb.\n";
        return;
    }

    const std::string area = stream.current_area_name();
    const auto schema  = dottalk::SchemaResolver::resolve(area);
    const auto sidecar = dottalk::SidecarLoader::load_json_sidecar(area);

    bool print_header_next = ps.status_verbose;
    PagerAction last_action = PagerAction::Repaint;
    bool last_nav_event = false;

    for (;;) {
        std::vector<dottalk::TupleRow> page;
        const bool should_fetch = (last_action == PagerAction::NextPage) || (print_header_next && ps.status_verbose);
        if (should_fetch) { auto tmp = stream.next_page(static_cast<std::size_t>(page_size)); page.swap(tmp); }

        if (ps.status_verbose && print_header_next) {
            auto* A = current_area(); std::string logical, base;
            try { if (A) { logical = A->logicalName(); base = A->dbfBasename(); } } catch (...) {}
            const std::string areaShow = logical.empty() ? "(none)" : logical + (base.empty() ? "" : (" [" + base + "]"));
            const std::string ord  = current_order_hint();
            const bool for_on = (stream.status_line().find("  FOR") != std::string::npos);
            std::cout << "HEADER: AREA=" << areaShow
                      << " | ORDER=" << ord
                      << " | FOR=" << (for_on ? "on" : "off")
                      << " | PAGE SIZE=" << page_size
                      << " | SPEC=" << (stream.current_spec().empty() ? "*" : stream.current_spec()) << "\n";
        }

        if (should_fetch) {
            if (page.empty()) {
                const bool for_on = (stream.status_line().find("  FOR") != std::string::npos);
                std::cout << "(end";
                if (for_on) std::cout << " — FOR active; try CLEAR FOR";
                else std::cout << " — no tuples; try SPEC " << stream.current_area_name() << ".*";
                std::cout << ")\n";
            } else {
                for (const auto& row : page) print_tuple_row(row);
            }
        }

        if (ps.show_schema) {
            std::cout << "Schema:\n";
            if (schema.fields.empty()) std::cout << "  (no logical schema)\n";
            else {
                std::cout << "  Source: " << schema.source << "\n";
                std::cout << "  Order: ";
                for (size_t i = 0; i < schema.fields.size(); ++i) { if (i) std::cout << " -> "; std::cout << schema.fields[i].name; }
                std::cout << "\n  Ownership: ";
                for (size_t i = 0; i < schema.fields.size(); ++i) { if (i) std::cout << ", "; std::cout << schema.fields[i].name << ":" << (schema.fields[i].kind.empty() ? "BASE" : schema.fields[i].kind); }
                std::cout << "\n";
            }
            for (const auto& w : schema.warnings) std::cout << "  WARNING: " << w << "\n";
        }

        if (ps.show_json) {
            std::cout << "JSON/sidecar:\n";
            if (!sidecar.has_value()) std::cout << "  (none)\n";
            else std::cout << "  " << *sidecar << "\n";
        }

        std::cout << stream.status_line() << "\n";

        auto pr = pager_prompt_and_dispatch(stream, ps);
        if (pr.action == PagerAction::Quit) break;

        if (auto h2 = dottalk::exprglue::pop_ambiguity_hint()) {
            std::cout << "WARNING: '" << *h2 << "' is ambiguous; use AREA.FIELD\n";
        }

        last_action = pr.action;
        last_nav_event = pr.nav_event;
        print_header_next = ps.status_verbose && pr.nav_event;
    }
}

} // namespace

void cmd_SUPER_BROWSER(xbase::DbArea& /*A*/, std::istringstream& iss) { run_super_browser(iss); }
void cmd_SUPERBROWSER(xbase::DbArea& A, std::istringstream& iss) { cmd_SUPER_BROWSER(A, iss); }
