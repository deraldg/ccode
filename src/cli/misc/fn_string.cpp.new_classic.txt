// Core FoxPro 2.6-compatible string functions
// All return std::string and take vector<string> exactly like UPPER/LOWER

// ALLTRIM(c) → removes leading and trailing spaces
static std::string dt_alltrim(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    const std::string s = args[0];
    size_t start = s.find_first_not_of(" \t");
    if (start == std::string::npos) return {};           // all whitespace
    size_t end = s.find_last_not_of(" \t");
    return s.substr(start, end - start + 1);
}

// LTRIM(c) → removes leading spaces only
static std::string dt_ltrim(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    const std::string s = args[0];
    size_t start = s.find_first_not_of(" \t");
    return (start == std::string::npos) ? std::string{} : s.substr(start);
}

// RTRIM(c) → removes trailing spaces only (FoxPro calls it TRIM() in later versions)
static std::string dt_rtrim(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    const std::string s = args[0];
    size_t end = s.find_last_not_of(" \t");
    return (end == std::string::npos) ? std::string{} : s.substr(0, end + 1);
}

// LEFT(c, n) → leftmost n characters
static std::string dt_left(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    const std::string s = args[0];
    int n = std::stoi(args[1]);
    if (n <= 0) return {};
    return s.substr(0, static_cast<size_t>(n) < s.size() ? n : s.size());
}

// RIGHT(c, n) → rightmost n characters
static std::string dt_right(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    const std::string s = args[0];
    int n = std::stoi(args[1]);
    if (n <= 0) return {};
    if (static_cast<size_t>(n) >= s.size()) return s;
    return s.substr(s.size() - n);
}

// SUBSTR(c, start [, len]) → substring, 1-based like FoxPro
static std::string dt_substr(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    const std::string s = args[0];
    int start = std::stoi(args[1]);                     // 1-based!
    if (start <= 0) start = 1;                          // FoxPro treats <=0 as 1
    size_t pos = static_cast<size_t>(start - 1);

    size_t len = s.size() - pos;                        // default: to end
    if (args.size() >= 3) {
        int requested = std::stoi(args[2]);
        if (requested > 0) len = static_cast<size_t>(requested);
    }
    if (pos >= s.size()) return {};
    return s.substr(pos, len);
}

// LEN(c) → length of string (including spaces, FoxPro exact behavior)
static std::string dt_len(const std::vector<std::string>& args) {
    if (args.empty()) return "0";
    return std::to_string(args[0].size());
}

// AT(cSearch, cText [, nOccurrence]) → position (1-based) of nth occurrence
static std::string dt_at(const std::vector<std::string>& args) {
    if (args.size() < 2) return "0";
    const std::string needle = args[0];
    const std::string haystack = args[1];
    int occurrence = (args.size() >= 3) ? std::stoi(args[2]) : 1;
    if (occurrence <= 0 || needle.empty()) return "0";

    size_t pos = haystack.find(needle, 0);
    int count = 0;
    while (pos != std::string::npos) {
        if (++count == occurrence) return std::to_string(pos + 1);  // 1-based
        pos = haystack.find(needle, pos + 1);
    }
    return "0";
}

// ATC(cSearch, cText [, nOccurrence]) → case-insensitive AT()
static std::string dt_atc(const std::vector<std::string>& args) {
    if (args.size() < 2) return "0";
    std::string needle = args[0];
    std::string haystack = args[1];
    int occurrence = (args.size() >= 3) ? std::stoi(args[2]) : 1;

    // Simple to-upper transformation for case-insensitivity
    std::transform(needle.begin(), needle.end(), needle.begin(),
        [](unsigned char c){ return std::toupper(c); });
    std::transform(haystack.begin(), haystack.end(), haystack.begin(),
        [](unsigned char c){ return std::toupper(c); });

    if (occurrence <= 0 || needle.empty()) return "0";

    size_t pos = haystack.find(needle, 0);
    int count = 0;
    while (pos != std::string::npos) {
        if (++count == occurrence) return std::to_string(pos + 1);
        pos = haystack.find(needle, pos + 1);
    }
    return "0";
}

// $ operator → "text1" $ "text2" → .T. if text1 is found anywhere in text2
static std::string dt_contains(const std::vector<std::string>& args) {
    if (args.size() < 2) return ".F.";
    return (args[1].find(args[0]) != std::string::npos) ? ".T." : ".F.";
}

// STUFF(c, start, nDelete, cInsert) → delete nDelete chars at start (1-based) and insert cInsert
static std::string dt_stuff(const std::vector<std::string>& args) {
    if (args.size() < 4) return {};
    std::string s = args[0];
    int start = std::stoi(args[1]);
    int del   = std::stoi(args[2]);
    const std::string ins = args[3];

    if (start <= 0) start = 1;
    size_t pos = static_cast<size_t>(start - 1);
    if (pos > s.size()) pos = s.size();

    if (del < 0) del = 0;
    size_t end = pos + static_cast<size_t>(del);
    if (end > s.size()) end = s.size();

    s.replace(s.begin() + pos, s.begin() + end, ins);
    return s;
}

// CHR(n) → character with ASCII code n
static std::string dt_chr(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    int n = std::stoi(args[0]);
    if (n < 0 || n > 255) return {};  // FoxPro limits to 0-255
    return std::string(1, static_cast<char>(n));
}

// ASC(c) → ASCII code of first character (or 0 if empty)
static std::string dt_asc(const std::vector<std::string>& args) {
    if (args.empty() || args[0].empty()) return "0";
    return std::to_string(static_cast<unsigned char>(args[0][0]));
}

// SPACE(n) → string of n spaces
static std::string dt_space(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    int n = std::stoi(args[0]);
    if (n <= 0) return {};
    return std::string(static_cast<size_t>(n), ' ');
}

// REPLICATE(c, n) → repeat string n times (FoxPro name) aka STRREPEAT
static std::string dt_replicate(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    const std::string s = args[0];
    int n = std::stoi(args[1]);
    if (n <= 0 || s.empty()) return {};

    std::string result;
    result.reserve(s.size() * static_cast<size_t>(n));
    for (int i = 0; i < n; ++i) result += s;
    return result;
}

// Existing ones you already had — kept unchanged but re-commented
static std::string dt_upper(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    std::string s = args[0];
    for (auto& c : s) c = static_cast<char>(std::toupper(static_cast<unsigned char>(c)));
    return s;
}

static std::string dt_lower(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    std::string s = args[0];
    for (auto& c : s) c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
    return s;
}

// STR(numeric, length [, decimals]) → right-justified string with leading spaces
// Exact FoxPro 2.6 rules: rounds (banker’s rounding in 2.6), no sign for positive, sign eats a digit
static std::string dt_str(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    double value = std::stod(args[0]);

    int len = 10;      // FoxPro defaults
    int dec = 0;
    if (args.size() >= 2) len = std::stoi(args[1]);
    if (args.size() >= 3) dec = std::stoi(args[2]);

    if (len < 1) len = 1;
    if (dec < 0) dec = 0;
    if (dec >= len) dec = len - 1;

    // Use stringstream with fixed precision — mimics FoxPro exactly
    std::ostringstream oss;
    oss << std::fixed << std::setprecision(dec) << value;

    std::string s = oss.str();

    // Remove the + sign FoxPro never shows for positive numbers
    if (s[0] == '+') s.erase(0, 1);

    // FoxPro rounds, then right-justifies with leading spaces
    // If number is too big → asterisks (like FoxPro!)
    if (static_cast<int>(s.length()) > len) {
        return std::string(static_cast<size_t>(len), '*');
    }

    // Pad left with spaces
    if (static_cast<int>(s.length()) < len) {
        s = std::string(static_cast<size_t>(len - s.length()), ' ') + s;
    }
    return s;
}

// VAL(c) → character → numeric (stops at first non-digit, ignores leading spaces)
static std::string dt_val(const std::vector<std::string>& args) {
    if (args.empty() || args[0].empty()) return "0";

    std::string s = args[0];
    size_t start = s.find_first_not_of(" \t");
    if (start != std::string::npos) s = s.substr(start);

    // Find first non-numeric (respect decimal point and sign)
    size_t i = 0;
    if (i < s.size() && (s[i] == '+' || s[i] == '-')) ++i;
    while (i < s.size() && (std::isdigit(static_cast<unsigned char>(s[i])) || s[i] == '.')) ++i;

    if (i == 0) return "0";                     // nothing valid found
    std::string numpart = s.substr(0, i);

    try {
        double d = std::stod(numpart);
        // FoxPro returns integer if no decimal part
        if (d == std::floor(d)) {
            return std::to_string(static_cast<long long>(d));
        } else {
            return std::to_string(d);
        }
    } catch (...) {
        return "0";
    }
}

// PADL(expr, length [, padchar]) → left-pad (like FoxPro PADL())
static std::string dt_padl(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    std::string s = args[0];
    int target = std::stoi(args[1]);
    char pad = (args.size() >= 3 && !args[2].empty()) ? args[2][0] : ' ';

    if (target <= 0) return {};
    if (static_cast<int>(s.length()) >= target) return s.substr(0, target);

    return std::string(static_cast<size_t>(target - s.length()), pad) + s;
}

// PADR(expr, length [, padchar]) → right-pad (most common in reports!)
static std::string dt_padr(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    std::string s = args[0];
    int target = std::stoi(args[1]);
    char pad = (args.size() >= 3 && !args[2].empty()) ? args[2][0] : ' ';

    if (target <= 0) return {};
    if (static_cast<int>(s.length()) >= target) return s.substr(0, target);

    s.resize(static_cast<size_t>(target), pad);
    return s;
}

// PADC(expr, length [, padchar]) → center-pad
static std::string dt_padc(const std::vector<std::string>& args) {
    if (args.size() < 2) return {};
    std::string s = args[0];
    int target = std::stoi(args[1]);
    char pad = (args.size() >= 3 && !args[2].empty()) ? args[2][0] : ' ';

    if (target <= 0) return {};
    if (static_cast<int>(s.length()) >= target) return s.substr(0, target);

    int left = (target - static_cast<int>(s.length())) / 2;
    int right = target - s.length() - left;

    return std::string(static_cast<size_t>(left), pad) + s +
           std::string(static_cast<size_t>(right), pad);
}

// LIKE(cPattern, cString) → .T./.F. with ? and * wildcards (FoxPro exact rules)
static std::string dt_like(const std::vector<std::string>& args) {
    if (args.size() < 2) return ".F.";
    const std::string pattern = args[0];
    const std::string text    = args[1];

    // Simple recursive wildcard matcher (handles only ? and * — FoxPro 2.6 style)
    auto like_match = [](const std::string& pat, const std::string& str, size_t p = 0, size_t s = 0) -> bool {
        while (true) {
            if (p == pat.size()) return s == str.size();
            if (pat[p] == '*') {
                if (p + 1 == pat.size()) return true;  // trailing * matches all
                for (size_t i = s; i <= str.size(); ++i) {
                    if (like_match(pat, str, p + 1, i)) return true;
                }
                return false;
            }
            if (s == str.size()) return false;
            if (pat[p] != '?' && pat[p] != str[s]) return false;
            ++p; ++s;
        }
    };

    return like_match(pattern, text) ? ".T." : ".F.";
}

// PROPER(c) → Title Case (First letter of each word upper, rest lower)
static std::string dt_proper(const std::vector<std::string>& args) {
    if (args.empty()) return {};
    std::string s = args[0];
    bool newword = true;
    for (size_t i = 0; i < s.size(); ++i) {
        unsigned char c = static_cast<unsigned char>(s[i]);
        if (std::isalpha(c)) {
            if (newword) {
                s[i] = static_cast<char>(std::toupper(c));
                newword = false;
            } else {
                s[i] = static_cast<char>(std::tolower(c));
            }
        } else {
            newword = true;   // space, punctuation, etc. starts new word
        }
    }
    return s;
}

// TRANSFORM(expr, picture) → very common in reports (we implement basic @ masks only for now)
// Supported: @R (rest of string literal), @L (left justify), @C (center), @Z (zero suppress), 999... etc.
static std::string dt_transform(const std::vector<std::string>& args) {
    if (args.size() < 2) return args.empty() ? {} : args[0];
    std::string value = args[0];
    std::string pic   = args[1];

    // Very common case: numeric with picture
    if (value.find_first_not_of(" \t-0123456789.") == std::string::npos) {
        try {
            double d = std::stod(value);
            // Simple 99999.99 style handling
            size_t digits = pic.find_first_not_of("0123456789.Xx9#");
            if (digits != std::string::npos) {
                // Fall back to STR() style if complex
                return value;
            }
            std::ostringstream oss;
            oss << std::fixed;
            size_t decpos = pic.find('.');
            if (decpos != std::string::npos) {
                int prec = pic.size() - decpos - 1;
                oss << std::setprecision(prec);
            }
            oss << d;
            return oss.str();
        } catch (...) { /* fall through */ }
    }

    // Default: just return the value (good enough for now)
    return value;
}
