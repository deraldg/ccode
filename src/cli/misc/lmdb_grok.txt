void cmd_LMDB(DbArea& A, std::istringstream& S)
{
    std::string sub;
    if (!(S >> sub)) {
        std::cout << "LMDB <CREATE|REBUILD|STATUS|SETORDER|DROP>\n";
        return;
    }

    const std::string U = textio::up(sub);
    if (U == "CREATE") {
        // create empty LMDB container with same stem
        std::string path = A.dbfDir() + "/" + A.dbfBasename() + ".lmdb";
        if (A._idx->createLMDB(path)) {
            std::cout << "LMDB container created: " << path << "\n";
            relations_api::refresh_if_enabled();
        }
    }
    else if (U == "REBUILD") {
        if (A.isTableDirty()) {
            std::cout << "TABLE is DIRTY. COMMIT first.\n";
            return;
        }
        if (A._idx->rebuildLMDB(A)) {
            std::cout << "LMDB rebuild complete.\n";
            A.clearStaleFlags();   // you already have this
        }
    }
    else if (U == "STATUS") {
        A._idx->printLMDBStatus();
    }
    // etc.
}
