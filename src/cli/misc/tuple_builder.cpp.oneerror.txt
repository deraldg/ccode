std::string trim(std::string s) {
    auto issp = [](unsigned char c){ return std::isspace(c); };
    s.erase(s.begin(), std::find_if(s.begin(), s.end(), [&](unsigned char c){ return !issp(c); }));
    s.erase(std::find_if(s.rbegin(), s.rend(), [&](unsigned char c){ return !issp(c); }).base(), s.end());
    return s;
}

std::string up(std::string s) {
    std::transform(s.begin(), s.end(), s.begin(),
                   [](unsigned char c){ return static_cast<char>(std::toupper(c)); });
    return s;
}

// inline comment stripping inside tokens: "&&", "//", " (", " ;"
void strip_inline_comment(std::string& s) {
    auto cut_at = [&](std::size_t pos){ if (pos != std::string::npos) s.erase(pos); };
    std::size_t p;
    p = s.find("&&"); if (p != std::string::npos) { cut_at(p); return; }
    p = s.find("//"); if (p != std::string::npos) { cut_at(p); return; }
    p = s.find(" ("); if (p != std::string::npos) { cut_at(p); return; }
    p = s.find(" ;"); if (p != std::string::npos) { cut_at(p); return; }
}

bool area_is_open(const xbase::DbArea* a) {
    if (!a) return false;
    try { return a->isOpen(); } catch (...) { return false; }
}

const xbase::DbArea* resolve_area(std::size_t slot) {
    if (slot >= workareas::count()) return nullptr;
    return workareas::at(slot);
}

int current_slot() { return static_cast<int>(workareas::current_slot()); }

std::string basename_upper(std::string path) {
    std::replace(path.begin(), path.end(), '\\', '/');
    auto slash = path.find_last_of('/');
    if (slash != std::string::npos) path.erase(0, slash + 1);
    auto dot = path.find_last_of('.');
    if (dot != std::string::npos) path.erase(dot);
    return up(path);
}

std::string area_display_name_upper(int slot) {
    const xbase::DbArea* A = resolve_area(static_cast<std::size_t>(slot));
    if (!area_is_open(A)) return "";
    try { return basename_upper(A->name()); } catch (...) { return ""; }
}

// If TABLE buffering is enabled for an area, find the most-recent buffered value for (recno, field1).
// Returns true and sets out_val if an override exists.
bool get_buffer_override(int area0, int recno, int field1, std::string& out_val) {
    if (!dottalk::table::in_range(area0)) return false;
    if (!dottalk::table::is_enabled(area0)) return false;
    const auto& tb = dottalk::table::get_tb_const(area0);
    if (tb.empty()) return false;

    auto range = tb.changes.equal_range(recno);
    if (range.first == range.second) return false;

    bool found = false;
    int best_prio = -2147483647;
    std::string best;

    for (auto it = range.first; it != range.second; ++it) {
        const auto& ce = it->second;
        auto nv = ce.new_values.find(field1);
        if (nv == ce.new_values.end()) continue;
        if (!found || ce.priority >= best_prio) {
            found = true;
            best_prio = ce.priority;
            best = nv->second;
        }
    }

    if (!found) return false;
    out_val = best;
    return true;
}

struct NameResolve { int slot = -1; bool ambiguous = false; std::vector<int> matches; };

NameResolve resolve_slot_by_area_name_all(const std::string& areaNameUpper) {
    NameResolve r;
    const std::size_t n = workareas::count();

    for (std::size_t i = 0; i < n; ++i) {
        const xbase::DbArea* A = resolve_area(i);
        if (!area_is_open(A)) continue;

        bool matched = false;

        // Match by basename of file
        try {
            const std::string bn = basename_upper(A->name());
            if (!bn.empty() && bn == areaNameUpper) matched = true;
        } catch (...) {}

        if (matched) {
            r.matches.push_back(static_cast<int>(i));
        }
    }

    if (r.matches.size() == 1) {
        r.slot = r.matches[0];
    } else if (!r.matches.empty()) {
        r.ambiguous = true;
    }

    return r;
}

struct ParsedRef {
    bool ok = true;
    bool star = false;
    int  slot = -1;
    std::string field;
};

struct ParseResult {
    bool ok = true;
    std::string error;
    std::vector<ParsedRef> refs;
};

ParsedRef parse_one(std::string tok) {
    ParsedRef r;
    tok = trim(tok);
    strip_inline_comment(tok);
    tok = trim(tok);
    if (tok.empty()) { r.ok = false; return r; }

    if (tok == "*" || up(tok) == "ALL") {
        r.star = true;
        return r;
    }

    // forms:
    //   FIELD
    //   AREA.FIELD
    //   [SLOT].FIELD
    //   SLOT.FIELD
    {
        auto dot = tok.find('.');
        if (dot == std::string::npos) {
            r.slot = current_slot();
            r.field = tok;
            return r;
        }

        std::string left = trim(tok.substr(0, dot));
        std::string right = trim(tok.substr(dot + 1));
        if (right.empty()) { r.ok = false; return r; }

        // [N]
        if (!left.empty() && left.front() == '[' && left.back() == ']') {
            left = left.substr(1, left.size() - 2);
        }

        // numeric slot?
        bool numeric = !left.empty() && std::all_of(left.begin(), left.end(), [](unsigned char c){ return std::isdigit(c) != 0; });
        if (numeric) {
            r.slot = std::stoi(left);
            r.field = right;
            return r;
        }

        // by area name
        NameResolve nr = resolve_slot_by_area_name_all(up(left));
        if (nr.ambiguous) { r.ok = false; return r; }
        if (nr.slot < 0) { r.ok = false; return r; }
        r.slot = nr.slot;
        r.field = right;
        return r;
    }
}

ParseResult parse_tokens(const std::vector<std::string>& toks) {
    ParseResult out;

    for (auto tok : toks) {
        tok = trim(tok);
        if (tok.empty()) continue;
        ParsedRef r = parse_one(tok);
        if (!r.ok) {
            out.ok = false;
            out.error = "ERROR: invalid tuple token: '" + tok + "'";
            return out;
        }
        out.refs.push_back(std::move(r));
    }

    if (out.refs.empty()) {
        out.ok = true;
        return out;
    }

    return out;
}

std::vector<std::pair<int,std::string>> expand(const ParsedRef& r) {
    std::vector<std::pair<int,std::string>> out;
    if (r.star) {
        const int slot = current_slot();
        const xbase::DbArea* A = resolve_area(static_cast<std::size_t>(slot));
        if (!area_is_open(A)) return out;

        try {
            const auto& fs = A->fields();
            for (const auto& fd : fs) out.emplace_back(slot, fd.name);
        } catch (...) {}
        return out;
    }

    out.emplace_back(r.slot, r.field);
    return out;
}

int safe_recno(const xbase::DbArea* A) {
    if (!A) return 0;
    try { return static_cast<int>(A->recno()); } catch (...) { return 0; }
}

TupleBuildResult build_tuple_from_spec(const std::string& spec_in, const TupleBuildOptions& opt) {
    TupleBuildResult res;

    if (opt.refresh_relations) {
        relations_api::refresh_for_current_parent();
    }

    std::string spec = trim(spec_in);
    if (spec.empty()) spec = "*";

    // tokenize by comma
    std::vector<std::string> toks;
    { std::stringstream ss(spec); std::string tok; while (std::getline(ss, tok, ',')) toks.push_back(tok); }

    auto pr = parse_tokens(toks);
    if (!pr.ok) { res.ok = false; res.error = pr.error; return res; }

    std::vector<std::pair<int,std::string>> items;
    for (const auto& r : pr.refs) {
        auto v = expand(r);
        items.insert(items.end(), v.begin(), v.end());
    }

    if (items.empty()) { res.ok = true; res.row = TupleRow{}; return res; }

    TupleRow row;
    row.columns.reserve(items.size());
    row.values.reserve(items.size());

    std::unordered_set<int> touched;

    for (const auto& it : items) {
        const int slot = it.first;
        std::string fieldName = it.second;

        const xbase::DbArea* ar = resolve_area(static_cast<std::size_t>(slot));
        std::string canonicalField = fieldName;

        // Resolve canonical case by schema match
        if (area_is_open(ar)) {
            try {
                const auto& fs = ar->fields();
                const std::string WANT = up(fieldName);
                for (const auto& fd : fs) {
                    if (up(fd.name) == WANT) { canonicalField = fd.name; break; }
                }
            } catch (...) {}
        }

        // Column label
        std::string colName = canonicalField;
        if (opt.header_area_prefix) {
            const std::string areaName = area_display_name_upper(slot);
            if (!areaName.empty()) colName = areaName + "." + canonicalField;
        }

        // Value
        std::string val;
        bool have_area = area_is_open(ar);

        if (have_area) {
            try {
                // Prefer schema match (case-insensitive)
                bool matched = false;
                const std::string WANT = up(canonicalField);
                const auto& fs = ar->fields();
                for (const auto& fd : fs) {
                    if (up(fd.name) == WANT) {
                        val = xfg::getFieldAsString(*const_cast<xbase::DbArea*>(ar), fd.name);
                        matched = true;
                        break;
                    }
                }
                if (!matched) {
                    if (opt.strict_fields) {
                        res.ok = false;
                        std::ostringstream oss;
                        oss << "ERROR: field '" << fieldName << "' not found in area slot " << slot << ".";
                        res.error = oss.str();
                        return res;
                    }
                    val = xfg::getFieldAsString(*const_cast<xbase::DbArea*>(ar), canonicalField);
                }
            } catch (...) {
                val.clear();
            }
        } else {
            if (opt.strict_fields) {
                res.ok = false;
                std::ostringstream oss;
                oss << "ERROR: area slot " << slot << " is not open.";
                res.error = oss.str();
                return res;
            }
            val.clear();
        }

        // Overlay TABLE-buffered edits (preview) for this area+recno+field.
        if (have_area) {
            std::string ov;
            const int area0 = slot;
            const int recno = safe_recno(ar);
            int field1 = 0;

            try {
                const auto& fs2 = ar->fields();
                const std::string want = up(canonicalField);
                for (std::size_t fi = 0; fi < fs2.size(); ++fi) {
                    if (up(fs2[fi].name) == want) {
                        field1 = static_cast<int>(fi) + 1;
                        break;
                    }
                }
            } catch (...) {
                field1 = 0;
            }

            if (field1 > 0 && get_buffer_override(area0, recno, field1, ov)) {
                val = ov;
            }
        }

        row.columns.push_back(TupleColumn{colName, slot, canonicalField});
        row.values.push_back(val);

        if (have_area) touched.insert(slot);
    }

    // Provenance fragments (DBF-level for now).
    for (int slot : touched) {
        const xbase::DbArea* ar = resolve_area(static_cast<std::size_t>(slot));
        if (!area_is_open(ar)) continue;

        try {
            row.provenance.push_back("DBF:" + basename_upper(ar->name()));
        } catch (...) {}
    }

    res.ok = true;
    res.row = std::move(row);
    return res;
}
