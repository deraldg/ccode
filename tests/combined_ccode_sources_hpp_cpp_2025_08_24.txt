#ifndef TEXTIO_HPP
#define TEXTIO_HPP
#include <string>
#include <algorithm>
#include <cctype>
namespace textio {
inline std::string trim(const std::string& s) {
    auto a = s.begin();
    while (a!=s.end() && std::isspace((unsigned char)*a)) ++a;
    auto b = s.end();
    while (b!=a && std::isspace((unsigned char)*(b-1))) --b;
    return std::string(a,b);
}
inline std::string up(const std::string& s){
    std::string t=s; for(char& c:t) c=(char)std::toupper((unsigned char)c); return t;}
}
#endif
