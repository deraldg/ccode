#include "xbase.hpp"                   // <-- Needed for DbArea
#include "textio.hpp"
#include "order_state.hpp"
#include "xindex/simple_index.hpp"
#include <filesystem>
#include <iostream>
#include <sstream>

using namespace textio;
namespace fs = std::filesystem;

// LIST [ALL|<n>]
void cmd_LIST(DbArea& a, std::istringstream& in)
{
    // Parse LIST options (limit / ALL) and column widths.
    auto opt = parse_list_options(in);
    auto recw = calc_widths(a);

    print_header(a, recw);

    // If an index is active for this area, list rows in its order.
    if (orderstate::hasOrder(a)) {
        const std::string inxPath = orderstate::orderPath(a);
        xindex::SimpleIndex inx;
        std::string err;
        if (!inx.open(inxPath, &err)) {
            std::cout << "Failed to open index: " << inxPath
                      << " (" << err << ")\n";
            // fall through to physical order
        } else {
            int printed = 0;
            const auto& ents = inx.entries();
            for (const auto& e : ents) {
                if (!a.gotoRec(static_cast<int32_t>(e.recno))) continue;
                if (!a.readCurrent()) continue;
                if (a.isDeleted()) continue;
                print_row(a, recw);
                ++printed;
                if (!opt.all && opt.limit > 0 && printed >= opt.limit) break;
            }

            if (!opt.all) {
                std::cout << printed << " record(s) listed (limit "
                          << opt.limit << "). Use LIST ALL to show more.\n";
            } else {
                std::cout << printed << " record(s) listed.\n";
            }
            return; // don’t run the physical-order loop
        }
    }

    // Fallback: physical-order loop (existing behavior)
    int printed = 0;
    if (!a.gotoTop()) {
        std::cout << "(empty)\n";
        return;
    }
    do {
        if (!a.readCurrent()) continue;
        if (a.isDeleted()) continue;
        print_row(a, recw);
        ++printed;
        if (!opt.all && opt.limit > 0 && printed >= opt.limit) break;
    } while (a.skip(1));

    if (!opt.all) {
        std::cout << printed << " record(s) listed (limit "
                  << opt.limit << "). Use LIST ALL to show more.\n";
    } else {
        std::cout << printed << " record(s) listed.\n";
    }
}
